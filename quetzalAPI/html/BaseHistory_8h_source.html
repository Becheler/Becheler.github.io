146  template<typename Generator>
147  coord_type backward_kernel(coord_type const& x, time_type t, Generator& gen) const
148  {
149  --t;
150  assert(m_flows->flux_to_is_defined(x,t));
151  if( ! m_kernel->has_distribution(x, t))
152  {
153  m_kernel->set(x, t, make_backward_distribution(x, t));
154  }
155  return m_kernel->operator()(gen, x, t);
156  }

170  auto make_backward_distribution(coord_type const& x, time_type const& t) const
171  {
172  std::vector<double> weights;
173  std::vector<coord_type> support;
174  weights.reserve(m_flows->flux_to(x,t).size());
175  support.reserve(m_flows->flux_to(x,t).size());
176  for(auto const& it : m_flows->flux_to(x,t) )
177  {
178  support.push_back(it.first);
179  weights.push_back(static_cast<double>(it.second));
180  }
