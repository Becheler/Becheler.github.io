156  bool is_in_spatial_extent(GeographicCoordinates const& c) const {
157  bool is_in_spatial_extent = false;
158  if( c.lon() >= m_extent.lon_min() && c.lon() < m_extent.lon_max() &&
159  c.lat() <= m_extent.lat_min() && c.lat() > m_extent.lat_max() )

181  template<typename F, typename G>
182  void export_to_geotiff(F f, G g, time_type const& t1, time_type const& t2, std::string const& filename) const {
183 
184  int nBands = std::abs(t2 - t1);
185  assert(nBands >= 1);
186 
187  // Create a model dataset band
188  GDALDataset* p_sink = get().GetDriver()->Create(filename.c_str(), width(), height(), nBands, GDT_Float32, NULL);
189  std::vector<float> buffer(width()*height());
190 
191  GDALRasterBand* p_source_band = get().GetRasterBand(1);
192  double na = p_source_band->GetNoDataValue();
193 
194  // Modify data
195  for(int i = 1; i <= nBands; ++i ){
196 
197 
198  auto err = p_source_band->RasterIO( GF_Read, 0, 0, width(), height(), buffer.data(), width(), height(), GDT_Float32, 0, 0 );
199  (void)err;
200 
201  //replace all continental cells value by 0
202  auto continental_cells = get_domain(0);
203  for(auto const& it : continental_cells){
204  buffer.at(it.x + it.y*width()) = 0.0;
205  }
206 
207  // Replace all cells where F is defined by F
208  auto t_curr = t1 + i-1;
209  auto f_defined_cells = g(t_curr);
210 
211  for(coord_type const& it: f_defined_cells){
212  auto xy = to_xy(it);
213  buffer.at(xy.x + (xy.y*width()) ) = static_cast<float>(f(it, t_curr));
214  }
215 
216  //write the data
217  GDALRasterBand* p_sink_band = p_sink->GetRasterBand(i);
218  p_sink_band->SetNoDataValue(na);
219  auto err2 = p_sink_band->RasterIO( GF_Write, 0, 0, width(), height(), buffer.data(), width(), height(), GDT_Float32, 0, 0 );
220  (void)err2;
221  p_sink->FlushCache();
222 
223  }
224 
225  // write metadata
226  std::vector<double> geo_transform(6);
227  get().GetGeoTransform( geo_transform.data() );
228  p_sink->SetGeoTransform( geo_transform.data() );
229  p_sink->SetProjection( get().GetProjectionRef() );
230 
231  GDALClose( (GDALDatasetH) p_sink );
232  }

234  void export_to_shapefile(std::map<coord_type, unsigned int> counts, std::string const& filename) const {
235  std::string driver_name = "ESRI Shapefile";
236  GDALDriver *poDriver;
237  GDALAllRegister();
238  poDriver = GetGDALDriverManager()->GetDriverByName(driver_name.c_str() );
239  assert( poDriver != NULL);
240 
241  GDALDataset *poDS;
242  poDS = poDriver->Create( filename.c_str(), 0, 0, 0, GDT_Unknown, NULL );
243  assert( poDS != NULL );
244 
245  OGRLayer *poLayer;
246  poLayer = poDS->CreateLayer( "sample", NULL, wkbPoint, NULL );
247  assert( poLayer != NULL );
248 
249  OGRFieldDefn oField( "count", OFTInteger );
250  if( poLayer->CreateField( &oField ) != OGRERR_NONE ){
251  throw(std::string("Creating sample size field failed."));
252  }
253 
254  for(auto const& it: counts){
255  double x = it.first.lon();
256  double y = it.first.lat();
257  int sample_size = it.second;
258 
259  OGRFeature *poFeature;
260  poFeature = OGRFeature::CreateFeature( poLayer->GetLayerDefn() );
261 
262  poFeature->SetField( "count", sample_size );
263 
264  OGRPoint pt;
265  pt.setX( x );
266  pt.setY( y );
267 
268  poFeature->SetGeometry( &pt );
269 
270  if( poLayer->CreateFeature( poFeature ) != OGRERR_NONE ){
271  throw(std::string("Failed to create feature in shapefile."));
272  }
273  OGRFeature::DestroyFeature( poFeature );
274  }
275  GDALClose( poDS );
276  }

359  // Read 1 value using GDAL API. To change if access is too slow
360  double read(unsigned int x, unsigned int y, unsigned int bandID) const {
361  int nXSize = 1;
362  auto line = (float *) CPLMalloc(sizeof(float)*nXSize);
363  auto err = band(bandID).RasterIO( GF_Read, x, y, 1, 1, line, nXSize, 1, GDT_Float32, 0, 0 );
364  if(err != CE_None){
365  throw std::invalid_argument( "received negative x y or band ID" );
366  };
367  double val = double(*line);
368  CPLFree(line);
369  return(val);
370  }

372  // Convert row x and col y to geographic coordinate
373  GeographicCoordinates to_lat_lon(unsigned int x, unsigned int y) const {
374  auto lon = m_gT[1] * x + m_gT[2] * y + m_gT[1] * 0.5 + m_gT[2] * 0.5 + m_gT[0];
375  auto lat = m_gT[4] * x + m_gT[5] * y + m_gT[4] * 0.5 + m_gT[5] * 0.5 + m_gT[3];
376  return GeographicCoordinates(lat, lon);
377  }

404  bool has_same_domain_at_all_times(){
405  bool answer = true;
406  if(depth()>1){
407  std::vector<unsigned int> bands(depth()-1);
408  std::iota(bands.begin(), bands.end(), 1);
409  auto s_0 = get_domain(0);
410  auto have_same_domain = [this, &s_0](unsigned int i){
411  auto s_i = this->get_domain(i);
412  return std::equal(s_0.begin(), s_0.end(), s_i.begin());
413  };
414  answer = std::all_of(bands.begin(), bands.end(), have_same_domain);
415  }
416  return answer;
417  }
