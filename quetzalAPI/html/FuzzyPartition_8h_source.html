115  FuzzyPartition & merge_clusters(RestrictedGrowthString const& rgs)
116  {
117  assert( rgs.size() == this->nClusters() );
118  matrix_type new_coefficients(rgs.nBlocks(), this->nElements(), 0);
120  for(unsigned int i = 0; i < m_membership_coefficients.size1(); ++i){

171  template<typename V, typename W>
172  double compute_cost(V const& v, W const& w) const {
173  assert(v.size() == w.size());
174 
175  std::vector<double> distances;
176 
177  for(auto && t : zip_range(v, w) ) {
178  distances.push_back(std::abs( boost::get<0>(t) - boost::get<1>(t) ));
179  }
180 
181  return std::accumulate(distances.begin(), distances.end(), 0.0);
182  }

193  template<typename T>
194  double compute_solution_cost(Matrix<T> const& solution, Matrix<T> const& costs) const {
195  double total_cost = 0.0;
196  for (unsigned int i = 0 ; i < solution.rows(); ++i ) {
197  int row_matches = 0;
198  for ( unsigned int j = 0 ; j < solution.columns() ; ++j ) {
199  if ( solution(i,j) == 0 ){
200  ++row_matches;
201  total_cost += costs(i,j);
202  }
203  }
204  assert(row_matches == 0 || row_matches == 1);
205  }
206  return total_cost;
207  }

227 
228  unsigned int n_rows = coefficients.begin()->second.size();
229  unsigned int n_cols = coefficients.size();

237  unsigned int i = 0;
