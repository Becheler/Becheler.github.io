256  km distanceEarth(double lat1d, double lon1d, double lat2d, double lon2d) const {
257 
258  double earthRadiusKm = 6371.0;
259  double lat1r, lon1r, lat2r, lon2r, u, v;
260 
261  lat1r = deg2rad(lat1d);
262  lon1r = deg2rad(lon1d);
263  lat2r = deg2rad(lat2d);
264  lon2r = deg2rad(lon2d);
265 
266  u = sin((lat2r - lat1r)/2);
267  v = sin((lon2r - lon1r)/2);
268 
269  km d = 2.0 * earthRadiusKm * asin(sqrt(u * u + cos(lat1r) * cos(lat2r) * v * v));
270 
271  assert(d >= 0);
272 
273  return d;
274  }
