208  km distanceEarth(double lat1d, double lon1d, double lat2d, double lon2d) const {
209 
210  double earthRadiusKm = 6371.0;
211  double lat1r, lon1r, lat2r, lon2r, u, v;
212 
213  lat1r = deg2rad(lat1d);
214  lon1r = deg2rad(lon1d);
215  lat2r = deg2rad(lat2d);
216  lon2r = deg2rad(lon2d);
217 
218  u = sin((lat2r - lat1r)/2);
219  v = sin((lon2r - lon1r)/2);
220 
221  km d = 2.0 * earthRadiusKm * asin(sqrt(u * u + cos(lat1r) * cos(lat2r) * v * v));
222 
223  assert(d >= 0);
224 
225  return d;
226  }
