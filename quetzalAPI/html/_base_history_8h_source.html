168  auto make_backward_distribution(coord_type const& x, time_type const& t) const
169  {
170  std::vector<double> weights;
171  std::vector<coord_type> support;
172  weights.reserve(m_flows->flux_to(x,t).size());
173  support.reserve(m_flows->flux_to(x,t).size());
174  for(auto const& it : m_flows->flux_to(x,t) )
175  {
176  support.push_back(it.first);
177  weights.push_back(static_cast<double>(it.second));
178  }
