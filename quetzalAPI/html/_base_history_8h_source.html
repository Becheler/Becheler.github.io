26 /*!
27 * @brief Base class for spatially explicit and forward-in time population history simulators.
28 * @details CRTP design pattern: this class has to be specialized (inherited from) to define History classes with different
29 * behaviors (eg mass-based or individual-based).
30 *
31 *
32 * @tparam Space Demes identifiers.
33 * @tparam Time EqualityComparable, CopyConstructible.
34 * @tparam DispersalPolicy Policy use for populations dynamics simulation algorithms, see dispersal_policy
35 *
36 * @ingroup demography
37 *
38 */
39 template<typename Space, typename Time, typename DispersalPolicy, typename StoragePolicy>
40 class BaseHistory {

42 public:

68  /**
69  * @brief Constructor initializing the demographic database.
70  *
71  * @param x the coordinate of introduction
72  * @param t the introduction time
73  * @param N the population size at coordinate x at time t
74  *
75  * \section Example
76  * \snippet demography/test/History/History_test.cpp Example
77  * \section Output

80  BaseHistory(coord_type const& x, time_type const& t, typename dispersal_policy::value_type N):
81  m_sizes(std::make_unique<pop_sizes_type>()),
82  m_flows(std::make_unique<flow_type>()),
83  m_kernel(std::make_unique<backward_kernel_type>())

138  /**
139  * @brief Samples a coordinate from the backward-in-time transition matrix
140  *
141  * \details The transition matrix is computed from the demographic flows
142  * database. The returned coordinate object will basically answer the question:
143  * when an individual is found in \f$x\f$ at time \f$t\f$, where could it
144  * have been at time \f$t-1\f$ ?
145  */
146  template<typename Generator>
147  coord_type backward_kernel(coord_type const& x, time_type t, Generator& gen) const
148  {
149  --t;
150  assert(m_flows->flux_to_is_defined(x,t));
151  if( ! m_kernel->has_distribution(x, t))
152  {
153  m_kernel->set(x, t, make_backward_distribution(x, t));
154  }
155  return m_kernel->operator()(gen, x, t);
156  }
157 
158 protected:
159 
160  // Need to be accessed by the expand method
161  std::unique_ptr<pop_sizes_type> m_sizes;
162  std::unique_ptr<flow_type> m_flows;
163  std::vector<Time> m_times;
164 
165  // mutable because sampling backward kernel can update the state
166  mutable std::unique_ptr<backward_kernel_type> m_kernel;
167 
168 private:
169 
170  auto make_backward_distribution(coord_type const& x, time_type const& t) const
171  {
172  std::vector<double> weights;
173  std::vector<coord_type> support;
174  weights.reserve(m_flows->flux_to(x,t).size());
175  support.reserve(m_flows->flux_to(x,t).size());
176  for(auto const& it : m_flows->flux_to(x,t) )
177  {
178  support.push_back(it.first);
179  weights.push_back(static_cast<double>(it.second));
180  }
181  return discrete_distribution_type(std::move(support),std::move(weights));
182  }
183 
184 }; // end BaseHistory
