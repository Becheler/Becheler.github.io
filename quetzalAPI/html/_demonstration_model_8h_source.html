144  // Construct a forest at a locus from the given dataset
145  forest_type make_forest(dataset_type const& data, dataset_type::locus_ID_type locus) const {
146  forest_type forest;
147  for(auto const& x : data.get_sampling_points())
148  {
149  for(auto const& individual : data.individuals_at(x))
150  {
151  auto alleles = individual.alleles(locus);
152  if(alleles.first.get_allelic_state() > 0) forest.insert(x, std::vector<coord_type>(1, x));
153  if(alleles.second.get_allelic_state() > 0) forest.insert(x, std::vector<coord_type>(1, x));
154  }
155  }
156  return forest;
157  }

168  auto make_growth_expression(param_type const& param, history_type::N_type const& N) const
169  {
170  using quetzal::expressive::literal_factory;
171  using quetzal::expressive::use;
172  literal_factory<coord_type, time_type> lit;
173  auto r = lit(param.r());
174  auto K = lit(param.k());
175  auto N_cref = std::cref(N);
176  auto N_expr = use([N_cref](coord_type x, time_type t){return N_cref(x,t);});
177  auto g = N_expr*(lit(1)+r)/ (lit(1)+((r * N_expr)/K));
178  auto sim_N_tilde = [g](generator_type& gen, coord_type x, time_type t){
179  std::poisson_distribution<history_type::N_type::value_type> poisson(g(x,t));
180  return poisson(gen);
181  };
182  return sim_N_tilde;
183  }

185  auto fuzzifie(forest_type const& forest, locus_ID_type const& locus) const {
186  std::map<coord_type, std::vector<double>> coeffs;
187 
188  for(auto const& it : m_forests.at(locus).positions())
189  {
190  coeffs[it].resize(forest.nb_trees());
191  }
192 
193  unsigned int cluster_id = 0;
194  for(auto const& it1 : forest )
195  {
196  for(auto const& it2 : it1.second)
197  {
198  coeffs[it2][cluster_id] += 1;
199  }
200  cluster_id += 1;
201  }
202 
203  for(auto & it1 : coeffs){
204  double sum = std::accumulate(it1.second.begin(), it1.second.end(), 0.0);
205  assert(sum > 0.0);
206  for(auto & it2 : it1.second){
207  it2 = it2/sum;
208  }
209  }
210  return FuzzyPartition<coord_type>(coeffs);
211  }

214  unsigned int factorial(unsigned int n) const
215  {
216  if (n == 0)
217  return 1;
218  return n * factorial(n - 1);
219  }
220 
221  int countP(int n, int k) const
222  {
223  // Base cases
224  if (n == 0 || k == 0 || k > n)
225  return 0;
226  if (k == 1 || k == n)
227  return 1;
228 
229  // S(n+1, k) = k*S(n, k) + S(n, k-1)
230  return k*countP(n-1, k) + countP(n-1, k-1);
231  }
232 
233  int Bell(unsigned int n) const
234  {
235  int sum = 0;
236  for(unsigned int k = 1; k <= n; ++k)
237  {
238  sum += countP(n, k);
239  }
240  return sum;
241  }

243  double pdf(unsigned int k, unsigned int n) const
244  {
245  double a = static_cast<double>(factorial(k));
246  double b = static_cast<double>(Bell(n));
247  double e = 2.71828;
248  double p = std::pow(k,n)/(a*b*e);
249  return p;
250  }

252  auto sample_partition(unsigned int n, Generator& gen) const
253  {
254  if(m_distribs.find(n) == m_distribs.end()){
255  std::vector<double> w;
256  size_t size = 10000;
257  w.resize(size);
258  w[0] = 0;
259  for(unsigned int k = 1; k < size; ++k)
260  {
261  w[k] = pdf(k,n);
262  }
263  m_distribs[n] = std::discrete_distribution<unsigned int>(w.begin(), w.end());
264  }
265 
266  unsigned int K = m_distribs.at(n)(gen);
267 
268  std::uniform_int_distribution<> unif(1, K);
269  std::vector<unsigned int> Cs;
270  for(unsigned int i = 1; i <= n; ++i )
271  {
272  Cs.push_back(unif(gen));
273  }
274 
275  std::map<unsigned int, unsigned int> ids;
276  std::vector<unsigned int> rgs;
277  rgs.resize(Cs.size());
278 
279  unsigned int block_ID = 0;
280  for(unsigned int i = 0; i < Cs.size(); ++i){
281  auto it = ids.find(Cs[i]);
282  if( it != ids.end() ){
283  rgs[i] = it->second;
284  }else {
285  rgs[i] = block_ID;
286  ids[Cs[i]] = block_ID;
287  ++block_ID;
288  }
289  }
290  return RestrictedGrowthString(rgs);
291  }

293 public:
294  auto operator()(generator_type& gen, param_type const& param) const
295  {
296  simulator_type simulator(m_x0, m_t0, param.N0());
297 
298  // Dispersal patterns
299  using quetzal::demography::dispersal::DiscreteLocationKernelFactory;
300  auto kernel = DiscreteLocationKernelFactory::make<coord_type, quetzal::geography::GeographicCoordinates::km, DispersalKernel>(m_distances, param);
301 
302  auto light_kernel = [&kernel, param](auto& gen, coord_type x, time_type t){
303  return kernel(x, gen);
304  };
305 
306  // Growth patterns
307  auto growth = make_growth_expression(param, simulator.size_history());
308 
309  auto merge_binop = [](const tree_type &parent, const tree_type &child)
310  {
311  tree_type copy = parent;
312  copy.insert( copy.end(), child.begin(), child.end() );
313  return copy;
314  };
315 
316  result_type fps;
317 
318  std::map<coord_type, unsigned int> counts;
319  for(auto const& it : m_dataset->get_sampling_points()){
320  counts[it] = (m_dataset->size(it)) * 2;
321  }
322 
323  auto const& history = simulator.simulate_demography(counts, growth, light_kernel, m_sampling_time, gen);
324 
325  for(auto const& locus : m_dataset->loci() ){
326  auto updated_forest = simulator.coalescence_process(m_forests.at(locus), history, merge_binop, gen);
327  auto S_sim = fuzzifie(updated_forest, locus);
328  //std::cout << "Simulated fuzzy Partiton:\n" << S_sim << std::endl;
329  if(S_sim.nClusters() > 1 )
330  {
331  S_sim.merge_clusters(sample_partition(S_sim.nClusters(), gen));
332  }
333  fps.push_back(S_sim);
334  }
335 
336  //std::cout << "Aggregated simulated fuzzy Partiton:\n" << S_sim << std::endl;
337 
338  return fps;
339 
340  }

343  FuzzyPartition<coord_type> fuzzifie_data(DemonstrationModel::dataset_type::locus_ID_type const& locus) const {
344  using cluster_type = unsigned int;
345  std::set<cluster_type> clusters;
346  std::map<coord_type, std::vector<double>> coeffs;
347 
348  auto frequencies = m_dataset->frequencies_discarding_NA(locus);
349  unsigned int n_clusters = m_dataset->allelic_richness(locus);
350 
351  // preparing coeffs clusters vectors
352  // and preparing clusters set
353  for(auto const& it1 : frequencies)
354  {
355  coeffs[it1.first].resize(n_clusters);
356  for(auto const& it2: it1.second){
357  clusters.insert(it2.first);
358  }
359  }
360 
361  // fill the coefficients
362  for(auto & it1 : coeffs){
363  for(auto const& it2 : frequencies[it1.first]){
364  auto allele = it2.first;
365  auto freq = it2.second;
366  auto it_pos = std::find(clusters.cbegin(), clusters.cend(), allele);
367  assert(it_pos != clusters.end());
368  auto index = std::distance(clusters.begin(), it_pos);
369  it1.second[index] = freq;
370  }
371  }
372 
373  return FuzzyPartition<coord_type>(coeffs);
374  }
