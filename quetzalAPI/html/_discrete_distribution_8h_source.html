20 /*!
21  * \brief Sampling (non) arithmetic values in discrete probability distribution.
22  *
23  * Produces random State objects, where the probability of each individual State
24  * \f$i\f$ is defined as \f$w = i/S\f$, that is the weight of the ith integer
25  * divided by the sum of all \f$n\f$ weights.
26  *
27  * \ingroup random
28  * \tparam State is the support of the probability distribution.
29  * \section Example
30  * \snippet random/test/DiscreteDistribution/DiscreteDistribution_test.cpp Example
31  * \section Output
32  * \include random/test/DiscreteDistribution/DiscreteDistribution_test.output
33  */

168  /** \brief Generates random objects that are distributed according to the associated parameter set. The entropy is acquired by calling g.operator().
169  * \tparam Generator must meet the requirements of <a href="http://en.cppreference.com/w/cpp/concept/UniformRandomBitGenerator">UniformRandomBitGenerator</a>.
170  * \param g an uniform random bit generator object
171  * \section Example
172  * \snippet random/test/DiscreteDistribution/DiscreteDistribution_test.cpp Example
173  * \section Output
174  * \include random/test/DiscreteDistribution/DiscreteDistribution_test.output
175  */
176  template<typename Generator>
177  result_type operator()(Generator& g) const
178  {
179  std::discrete_distribution <int> d(m_param.weights().cbegin(), m_param.weights().cend());
180  return m_param.support()[d(g)];
181  }
182 
183  /** \brief Generates random numbers that are distributed according to params
184  * The entropy is acquired by calling g.operator().
185  * \tparam Generator must meet the requirements of <a href="http://en.cppreference.com/w/cpp/concept/UniformRandomBitGenerator">UniformRandomBitGenerator</a>.
186  * \param g an uniform random bit generator object
187  * \param params distribution parameter set to use instead of the associated one
188  * \section Example
189  * \snippet random/test/DiscreteDistribution/DiscreteDistribution_test.cpp Example
190  * \section Output
191  * \include random/test/DiscreteDistribution/DiscreteDistribution_test.output
192  */
193  template<typename Generator>
194  result_type operator()(Generator& g, const param_type & params) const
195  {
196  std::discrete_distribution <int> d(params.weights().cbegin(), params.weights().cend());
197  return params.support()[d(g)];
198  }
