27 /*!
28  * \brief Discretized representation of a landscape and its ecological
29  * quantities through time and space
30  *
31  * A geographic explicit landscape is represented by a set of geographic coordinates
32  * at which are defined the values of a defined set of environmental quantities (typically bioclimatic variables).
33  * This geo-spatialized database is constructed from raster files and gives strong
34  * guarantees on data consistency.
35  *
36  * \tparam Key Ecological quantities identifier
37  * \tparam Time Time type
38  * \ingroup geography
39  * \section Example
40  * \snippet geography/test/DiscreteLandscape/DiscreteLandscape_test.cpp Example
41  * \section Output
42  * \include geography/test/DiscreteLandscape/DiscreteLandscape_test.output
43  */
44 template<typename Key, typename Time>
45 class DiscreteLandscape
46 {
47 
48  using geo_data_type = EnvironmentalQuantity<Time>;
49 
50 public:
51 
52  //! \typedef ecological quantities identifier type
53  using key_type = Key;
54 
55  //! \typedef coordinate type
56  using coord_type = typename geo_data_type::coord_type;
57 
58  //! \typedef time type
59  using time_type = Time;
60 
61  //! \typedef decimal degree type
62  using decimal_degree = typename geo_data_type::decimal_degree;
63 
64 private:
65 
66  std::map<key_type, geo_data_type> m_quantities;
67 
68  void build(std::map<key_type, std::string> const& files, std::set<Time> times){
69  for(auto const& it : files){
70  m_quantities.emplace( std::piecewise_construct,
71  std::forward_as_tuple(it.first),
72  std::forward_as_tuple(it.second, std::vector<Time>(times.begin(), times.end())));
73  }
74  }
75 
76 public:
77 
78  template<typename F, typename G>
79  void export_to_geotiff(F f, time_type const& t1, time_type const& t2, G g, std::string const& filename) const {
80  m_quantities.cbegin()->second.export_to_geotiff(f,g, t1, t2, filename);
81  }
82 
83  void export_to_shapefile(std::map<coord_type, unsigned int> counts, std::string const& filename) const {
84  m_quantities.cbegin()->second.export_to_shapefile(counts, filename);
85  }
86 
87  void export_to_shapefile(std::vector<coord_type> points, std::string const& filename) const {
88  m_quantities.cbegin()->second.export_to_shapefile(points, filename);
89  }

91  /**
92  * \brief Constructor
93  *
94  * \param files a map giving as keys the identifier of the ecological quantity,
95  * and as value a std::string giving the raster file to read.
96  * \param times the time values associated to each layer of the file: here a raster file represents
97  * an environmental quantity where the i-th layer represents the values across space at the i-th time.
98  * \remark error will be thrown if the datasets do not have the exactly same geographical properties
99  * (origin, extent, resolution, number of layers).
100  * \remark error will be thrown if data do not have same geographic definition space
101  * (coordinates of cells with defined value)
102  * \remark All quantities types are casted to double.
103  * \section Example
104  * \snippet geography/test/DiscreteLandscape/DiscreteLandscape_test.cpp Example
105  * \section Output
106  * \include geography/test/DiscreteLandscape/DiscreteLandscape_test.output
107  */
108  DiscreteLandscape(std::map<key_type, std::string> const& files, std::set<Time> const& times){
109 
110  build(files, times);
111 
112  auto have_same_origin = [this](auto const& it){return it.second.origin() == m_quantities.cbegin()->second.origin();};
113  if( ! all_of(m_quantities.cbegin(), m_quantities.cend(), have_same_origin)){
114  throw std::runtime_error("all ecological quantities dataset must have same origin");
115  }
116 
117  auto have_same_extent = [this](auto const& it){return it.second.extent() == m_quantities.cbegin()->second.extent();};
118  if( ! all_of(m_quantities.cbegin(), m_quantities.cend(), have_same_extent)){
119  throw std::runtime_error("all ecological quantities dataset must have same extent");
120  }
121 
122  auto have_same_resolution = [this](auto const& it){return it.second.resolution() == m_quantities.cbegin()->second.resolution();};
123  if( ! all_of(m_quantities.cbegin(), m_quantities.cend(), have_same_resolution)){
124  throw std::runtime_error("all ecological quantities dataset must have same resolution");
125  }
126 
127  auto have_same_depth = [this](auto const& it){return it.second.depth() == m_quantities.cbegin()->second.depth();};
128  if( ! all_of(m_quantities.cbegin(), m_quantities.cend(), have_same_depth)){
129  throw std::runtime_error("all ecological quantities dataset must have same depth");
130  }
131 
132  auto have_same_domain = [this](auto const& it){
133  auto const& domain = it.second.geographic_definition_space();
134  return std::equal(domain.cbegin(),
135  domain.cend(),
136  m_quantities.cbegin()->second.geographic_definition_space().cbegin());
137  };
138  if( ! all_of(m_quantities.cbegin(), m_quantities.cend(), have_same_domain)){
139  throw std::runtime_error("all ecological quantities dataset must have same domain");
140  }
141  }
142 
143  /**
144  * \brief Retrieves the number of ecological quantities.
145  * \return the number of ecological quantities.
146  * \section Example
147  * \snippet geography/test/DiscreteLandscape/DiscreteLandscape_test.cpp Example
148  * \section Output
149  * \include geography/test/DiscreteLandscape/DiscreteLandscape_test.output
150  */
151  auto quantities_nbr() const {
152  return m_quantities.size();
153  }
154 
155  /**
156  * \brief Accesses the specified ecological quantity
157  * \return a function-object with signature `value_type fun(coord_type const& x, time_type t)`
158  * which when inovked will return the value of the specified quantity.
159  * \param k the identifier of teh quantity to be retrieved
160  * \remark the function-obect is invalidated if the DiscreteLandscape is destroyed.
161  * \remark the returned lightweight object can be used for mathematical function composition using the expressive module.
162  * \section Example
163  * \snippet geography/test/DiscreteLandscape/DiscreteLandscape_test.cpp Example
164  * \section Output
165  * \include geography/test/DiscreteLandscape/DiscreteLandscape_test.output
166  */
167  auto operator [](key_type const& k) const {
168  assert(m_quantities.find(k) != m_quantities.end());
169  auto c = std::cref(m_quantities.at(k));
170  return [c](coord_type const& x, time_type t){return c.get().at(x,t); };
171  }
172 
173  /**
174  * \brief read the times at which quantity is defined.
175  * \return the ordered times that have been specified at construction
176  * \section Example
177  * \snippet geography/test/DiscreteLandscape/DiscreteLandscape_test.cpp Example
178  * \section Output
179  * \include geography/test/DiscreteLandscape/DiscreteLandscape_test.output
180  */
181  const std::vector<time_type>& temporal_definition_space() const {
182  return m_quantities.cbegin()->second.temporal_definition_space();
183  }

185  /**
186  * \brief The geographic coordinates of the centroids of the demes for which
187  * values are defined for all ecological quantities at all times.
188  *
189  * Returns the geographic coordinates of the centroids of the demes for which
190  * values are defined for all ecological quantities at all times.
191  * Values can be read from other coordinates, but this function provides a natural

197  * \include geography/test/DiscreteLandscape/DiscreteLandscape_test.output

200  return m_quantities.cbegin()->second.geographic_definition_space();
201  }

239  /**
240  * \brief Check if a given geographic coordinate is in the landscpae spatial extent
241  * \return true if `c` is in the landscape limits, else false
242  * \param c the coordinate to be tested
243  * \section Example
244  * \snippet geography/test/DiscreteLandscape/DiscreteLandscape_test.cpp Example
245  * \section Output
246  * \include geography/test/DiscreteLandscape/DiscreteLandscape_test.output
247  */
248  bool is_in_spatial_extent(coord_type const& c) const {
249  return m_quantities.cbegin()->second.is_in_spatial_extent(c);
250  }
251 
252  /**
253  * \brief Reprojects a coordinate to the centroid of the landscape cell to which it belongs
254  * \return the geograhic coordinate of the cell centroid
255  * \param c the coordinate to be reprojected
256  * \remark the given coordinate must be in the landscape spatial extent
257  * \section Example
258  * \snippet geography/test/DiscreteLandscape/DiscreteLandscape_test.cpp Example
259  * \section Output
260  * \include geography/test/DiscreteLandscape/DiscreteLandscape_test.output
261  */
262  coord_type reproject_to_centroid(coord_type const& c) const
263  {
264  return m_quantities.cbegin()->second.reproject_to_centroid(c);
265  }
266  /*!
267  \brief Get the four neighbors of a cell - or less if on the border of the grid
268  \param x0 the coordinates of the cell to get the four neighbors from
269  \pre x0 is in the landscape spatial extent
270  \post At least 2 neighbors (x0 is a corner) or 3 (x_0 on the landscape boundary)
271  \return A vector with no more than 4 coordinates.
272  */
273  std::vector<coord_type> direct_neighbors(coord_type const& x0) const {
274  assert(is_in_spatial_extent(x0));
275  auto res = resolution();
276  std::vector<coord_type> v;
277  coord_type x1(x0);
278  x1.lon() += res.lon();
279  if(is_in_spatial_extent(x1))
280  {
281  v.push_back(reproject_to_centroid(x1));
282  }
283  coord_type x2(x0);
284  x2.lat() += res.lat();
285  if(is_in_spatial_extent(x2))
286  {
287  v.push_back(reproject_to_centroid(x2));
288  }
289  coord_type x3(x0);
290  x3.lon() -= res.lon();
291  if(is_in_spatial_extent(x3))
292  {
293  v.push_back(reproject_to_centroid(x3));
294  }
295  coord_type x4(x0);
296  x4.lat() -= res.lat();
297  if(is_in_spatial_extent(x4))
298  {
299  v.push_back(reproject_to_centroid(x4));
300  }
301  assert(v.size() >= 2);
302  assert(v.size() <= 4);
303  return v;
304  }
305 };
