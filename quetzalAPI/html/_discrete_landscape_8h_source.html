262  std::vector<coord_type> four_nearest_defined_cells(coord_type const& x) const {
263  auto const& X = geographic_definition_space();
264 
265  auto x0 = reproject_to_centroid(x);
266  assert( std::find(X.begin(), X.end(), x0) != X.end());
267 
268  auto res = resolution();
269  std::vector<coord_type> v;
270 
271  coord_type x1(x0);
272  x1.lon() += res.lon();
273  if(is_in_spatial_extent(x1)){
274  x1 = reproject_to_centroid(x1);
275  if(std::find(X.begin(), X.end(), x1) != X.end()){
276  v.push_back(x1);
277  }
278  }
