42  /* @brief Coalesce a spatial forest of trees in a Wrigh-Fisher population until MRCA has been found.
43  *
44  * @tparam Space the coordinate type
45  * @tparam Tree the tree type
46  * @tparam Generator a random number Generator
47  * @tparam Binop a BranchingOperation functor for coalescence
48  * @tparam TimeFun an intialization functor for coalescence
49  *
50  * @param forest the spatial distribution of lineages to coalesce
51  * @param N number of gene copies in the Wright-Fisher population
52  * @param gen a random number Generator
53  * @param branch a branching binary operation (functor usually given by the selected coalescence policy).
54  * @param make_tree a initialization functor (functor usually given by the selected coalescence policy).
55  *
56  * @detail Coalesce lineages by sampling the waiting times directly in their distribution.
57  *
58  * @return One single tree.
59  */
60  template<typename Space, typename Tree, typename Generator, typename Binop, typename TimeFun>
61  static Tree coalesce(forest_type<Space, Tree> const& forest, unsigned int N, Generator& gen, Binop branch, TimeFun make_tree)
62  {
63 
64  if(forest.nb_trees() == 0){
65 
66  throw std::logic_error("The forest to coalesce is empty.");
67 
68  }else if(forest.nb_trees()==1){
69 
70  auto range = forest.trees_at_same_position(*(forest.positions().begin()));
71  assert(std::distance(range.first, range.second) == 1);
72  return range.first->second;
73 
74  }
75  assert(forest.nb_trees() >= 2 && "Trying to coalesce less than 2 nodes.");
76  auto trees = forest.get_all_trees();
77  return coalesce(trees, N, gen, branch, make_tree);
78  }

80  /* @brief Simulates the number of generations to wait before the next coalescence event
81  * @tparam Generator a random number Generator
82  * @param k number of lineages
83  * @param N number of gene copies in the Wright-Fisher population
84  * @param gen a random number Generator
85  * @detail Samples the number of generations to wait before the next coalescence
86  * event in a geometric distribution of parameter {k \choose 2} / N
87  * (see Molecular evolution, a statistical approach, Z. Yang, 9.2 p 313).
88  * @return the number of generations to wait before the next coalescence event.
89  */
90  template<typename Generator>
91  static unsigned int sample_waiting_time(unsigned int k, unsigned int N, Generator& gen)
92  {
93  assert(N > k);
94  double mean = static_cast<double>(N) / boost::math::binomial_coefficient<double>(k, 2);
95  double p = 1 / (1 + mean);
96  assert( 0 < p && p <= 1);
97  std::geometric_distribution dist(p);
98  return dist(gen);
99  }
100  /*
101  * @brief Coalesce a non-spatial forest of trees in a Wrigh-Fisher population until MRCA has been found.
102  *
103  * @tparam Tree the tree type
104  * @tparam Generator a random number Generator
105  * @tparam Binop a BranchingOperation functor for coalescence
106  * @tparam TimeFun an intialization functor for coalescence
107  *
108  * @param trees the set of lineages to coalesce
109  * @param N number of gene copies in the Wright-Fisher population
110  * @param gen a random number Generator
111  * @param branch a branching binary operation (functor usually given by the selected coalescence policy).
112  * @param make_tree a initialization functor (functor usually given by the selected coalescence policy).
113  *
114  * @detail Coalesce lineages by sampling the waiting times directly in their
115  * distribution. The algorithm will perform one generation of simultaneous
116  * multiple merges if the number of lineages to coalesce k is superior to the
117  * population size N.
118  *
119  * @return One single tree.
120  */
121  template<typename Tree, typename Generator, typename Binop, typename TimeFun>
122  static Tree coalesce(std::vector<Tree> & trees, unsigned int N, Generator& gen, Binop branch, TimeFun make_tree)
123  {
124  assert(trees.size() >= 2 && "Trying to coalesce less than 2 nodes.");
125  if (trees.size() > N)
126  {
127  // Wright Fisher assumptions are clearly violated
128  // Let's do one generation of simultaneous multiple merges to fix it.
129  unsigned int g = 1;
130  using smm_type = coalescence::SimultaneousMultipleMerger<coalescence::occupancy_spectrum::on_the_fly>;
131  auto last = smm_type::merge(trees.begin(), trees.end(), N, make_tree(g), branch, gen);
132  trees.erase(last, trees.end());
133  }
134  unsigned int k = trees.size();
135  assert( k <= N && "number of lineages greater than ancestral Wright-Fisher population size.");
136  auto last = trees.end();
137  while( k > 1 ){
138  unsigned int g = sample_waiting_time(k, N, gen);
139  last = quetzal::coalescence::binary_merge(trees.begin(), last, make_tree(g), branch, gen);
140  --k;
141  }
142  return *(trees.begin());
143  }
144  /*
145  * @brief Coalesce a non-spatial forest of trees in a single Wrigh-Fisher population
146  * during a finite number of generations.
147  *
148  * @tparam Tree the tree type
149  * @tparam Generator a random number Generator
150  * @tparam Binop a BranchingOperation functor for coalescence
151  * @tparam TimeFun an intialization functor for coalescence
152  *
153  * @param trees the set of lineages to coalesce
154  * @param N number of gene copies in the Wright-Fisher population
155  * @param gen a random number Generator
156  * g the number of generations that lasts the coalescence process
157  * @param branch a branching binary operation (functor usually given by the selected coalescence policy).
158  * @param make_tree a initialization functor (functor usually given by the selected coalescence policy).
159  *
160  * @detail Coalesce lineages by sampling the waiting times directly in their
161  * distribution. The algorithm will perform one generation of simultaneous
162  * multiple merges if the number of lineages to coalesce k is superior to the
163  * population size N.
164  *
165  * @return A set of trees, possibly of length 1 if MRCA was found during the g generations.
166  */
167  template<
168  typename MergerType = coalescence::SimultaneousMultipleMerger<coalescence::occupancy_spectrum::on_the_fly>,
169  typename Tree, typename Generator, typename Binop, typename TimeFun>
170  static std::vector<Tree> coalesce(std::vector<Tree> & trees, unsigned int N, unsigned int g, Generator& gen, Binop branch, TimeFun make_tree)
171  {
172  assert(trees.size() >= 2 && "Trying to coalesce less than 2 nodes.");
173  assert(g > 0 && "Number of generations for the coalescence process should be at least 1");
174  assert(N >= 1 && "Population size should be positive for evaluating coalescence probability");
175  using merger_type = MergerType;
176  unsigned int t = 0;
177  auto last = trees.end();
178  while( t < g && std::distance(trees.begin(), last) > 1)
179  {
180  last = merger_type::merge(trees.begin(), last, N, make_tree(t), branch, gen);
181  ++t;
182  }
183  trees.erase(last, trees.end());
184  return trees;
185  }
