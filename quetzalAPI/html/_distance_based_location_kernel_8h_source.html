22 // contructed at once
23 template<typename Space>
24 class DiscreteLocationTransitionKernel
25 {
26  using coord_type = Space;
27  std::vector<coord_type> m_points;
28 
29  using ID_type = unsigned int;
30  std::unordered_map<coord_type, ID_type> m_dico;
31 
32  using distribution_type = std::discrete_distribution<ID_type>;
33  mutable std::vector<distribution_type> m_kernels;
34 
35  std::unordered_map<coord_type, unsigned int> make_dico(std::vector<coord_type> const& points) const
36  {
37  unsigned int position = 0;
38  std::unordered_map<coord_type, unsigned int> m;
39  for(auto const& it : points)
40  {
41  m.emplace(it, position);
42  ++position;
43  }
44  return m;
45  }
46 
47 public:
48 
49  DiscreteLocationTransitionKernel(std::vector<coord_type> const& points, std::vector<distribution_type> && kernels):
50  m_points(points),
51  m_dico(make_dico(points)),
52  m_kernels(kernels)
53  {}
54 
55  template<typename Generator>
56  coord_type operator()(coord_type const& x, Generator & gen) const
57  {
58  auto id = m_dico.at(x);
59  auto & dist = m_kernels.at(id);
60  return m_points.at(dist(gen));
61  }
62 
63 };
64 
65 struct DiscreteLocationKernelFactory {
66 
67  template<typename Space, typename Value, typename Kernel>
68  static auto make(SymmetricDistanceMatrix<Space, Value> const& d, typename Kernel::param_type const& p)
69  {
70  Kernel k;
71  auto f = [k,p](Value r){return k.pdf(r, p);};
72  auto weights = d.apply(f);
73  std::vector<std::discrete_distribution<unsigned int>> discrete_kernels;
74  discrete_kernels.reserve(weights.points().size());
75  for(auto const& x : weights.points())
76  {
77  auto x_weights = weights.at(x);
78  discrete_kernels.push_back(std::discrete_distribution<unsigned int>(x_weights.begin(), x_weights.end()));
79  }
80  return DiscreteLocationTransitionKernel<Space>(weights.points(), std::move(discrete_kernels));
81  }
82 
83 };
