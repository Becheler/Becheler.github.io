204  bool has_same_domain_at_all_times(){
205  bool answer = true;
206  if(depth()>1){
207  std::vector<unsigned int> bands(depth()-1);
208  std::iota(bands.begin(), bands.end(), 1);
209  auto s_0 = get_domain(0);
210  auto have_same_domain = [this, &s_0](unsigned int i){
211  auto s_i = this->get_domain(i);
212  return std::equal(s_0.begin(), s_0.end(), s_i.begin());
213  };
214  answer = std::all_of(bands.begin(), bands.end(), have_same_domain);
215  }
216  return answer;
217  }

221  double read(unsigned int x, unsigned int y, unsigned int bandID) const {
222  int nXSize = 1;
223  auto line = (float *) CPLMalloc(sizeof(float)*nXSize);

232  // Convert row x and col y to geographic coordinate
233  GeographicCoordinates to_lat_lon(unsigned int x, unsigned int y) const {
234  auto lon = m_gT[1] * x + m_gT[2] * y + m_gT[1] * 0.5 + m_gT[2] * 0.5 + m_gT[0];
235  auto lat = m_gT[4] * x + m_gT[5] * y + m_gT[4] * 0.5 + m_gT[5] * 0.5 + m_gT[3];
236  return GeographicCoordinates(lat, lon);
237  }
