247  Extent<decimal_degree> compute_extent(GeographicCoordinates const& o, Resolution<decimal_degree> const& r) const {
248  auto lon_max = o.lon() + width() * r.lon();

250  return Extent<decimal_degree>(o.lat(), lat_max, o.lon(), lon_max);

253  std::vector<XY> get_domain(unsigned int bandID){
254  assert( bandID < depth());
255  std::vector<XY> v;
256  for(unsigned int x = 0; x < width(); ++x){
257  for( unsigned int y = 0; y < height(); ++y){
258  if(std::isfinite(read(x, y, bandID))){
259  v.emplace_back(x,y);
260  }
261  }
262  }
263  return v;
264  }

266  bool has_same_domain_at_all_times(){
267  bool answer = true;
268  if(depth()>1){
269  std::vector<unsigned int> bands(depth()-1);
270  std::iota(bands.begin(), bands.end(), 1);
271  auto s_0 = get_domain(0);
272  auto have_same_domain = [this, &s_0](unsigned int i){
273  auto s_i = this->get_domain(i);
274  return std::equal(s_0.begin(), s_0.end(), s_i.begin());
275  };
276  answer = std::all_of(bands.begin(), bands.end(), have_same_domain);
277  }
278  return answer;
279  }

282  // Read 1 value using GDAL API. To change if access is too slow
283  double read(unsigned int x, unsigned int y, unsigned int bandID) const {
284  int nXSize = 1;
285  auto line = (float *) CPLMalloc(sizeof(float)*nXSize);
286  auto err = band(bandID).RasterIO( GF_Read, x, y, 1, 1, line, nXSize, 1, GDT_Float32, 0, 0 );
287  assert(err == CE_None);
288  double val = double(*line);
289  CPLFree(line);
290  return(val);
291  }

294  // Convert row x and col y to geographic coordinate
295  GeographicCoordinates to_lat_lon(unsigned int x, unsigned int y) const {
296  auto lon = m_gT[1] * x + m_gT[2] * y + m_gT[1] * 0.5 + m_gT[2] * 0.5 + m_gT[0];
297  auto lat = m_gT[4] * x + m_gT[5] * y + m_gT[4] * 0.5 + m_gT[5] * 0.5 + m_gT[3];
298  return GeographicCoordinates(lat, lon);
299  }
300 
301  // Convert latitude/longitude to col/row indices
302  XY to_xy(GeographicCoordinates const& c) const {
303  int col = (c.lon() - m_gT[0]) / m_gT[1];
304  int row = (c.lat() - m_gT[3]) / m_gT[5];
305  return XY(col, row);
306  }
307 
308  std::vector<GeographicCoordinates> to_lat_lon(std::vector<XY> const& xy) const {
309  std::vector<GeographicCoordinates> out;
310  for(auto const& it : xy){
311  out.push_back(to_lat_lon(it.x, it.y));
