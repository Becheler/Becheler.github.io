389  bool has_same_domain_at_all_times(){
390  bool answer = true;

393  std::iota(bands.begin(), bands.end(), 1);
394  auto s_0 = get_domain(0);

396  auto s_i = this->get_domain(i);
397  return std::equal(s_0.begin(), s_0.end(), s_i.begin());
398  };
399  answer = std::all_of(bands.begin(), bands.end(), have_same_domain);
400  }
