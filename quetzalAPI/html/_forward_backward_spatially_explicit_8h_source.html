24  /**
25  * @brief Discrete-time coalescence simulator in a discrete spatially explicit landscape.
26  *
27  * @details This simulator consists in a forward demographic simulation of population
28  * growth and migration, after what a backward coalescent simulation is performed
29  * that traces back the ancestry of a sample of gene copies sampled in one
30  * or more populations. Most Recent Common ancestor is not always fond in the
31  * time span of the spatial history, so if the chosen CoalescencePolicy implements
32  * the required behavior, ancestry can be simulated in the more distant past using another model
33  * like the DiscreteTimeWrightFisher model, to find the MRCA.
34  * Details concerning the mergers (binary mergers, simultaneous multiple mergers)
35  * can be controlled using the relevant Merger policy class when calling a
36  * coalescence function of the class interface.
37  * @tparam Space deme identifiers (like the populations geographic coordinates)
38  * @tparam Time time identifier (like an integer representing the year)
39  * @tparam Strategy a template class argument (e.g. mass_based, individual_based)
40  * that indicates how dispersal should be defined in terms of
41  * demographic expansion algorithms. Strategy class
42  * implements the details (computing migrations probabilites as a function of the
43  * distance for example) whereas template specialization of the
44  * internal history_type class defines different migration algorithms.
45  * @tparam CoalescencePolicy a policy class for the demographic expansion.
46  *
47  * @ingroup simulator
48  *
49  */
50  template<typename Space, typename Time, typename DispersalPolicy, typename CoalescencePolicy, typename StoragePolicy>
51  class ForwardBackwardSpatiallyExplicit :
52  public CoalescencePolicy
53  {
54  public:
55  //! \typedef space type
56  using coord_type = Space;
57  //! \typedef time type
58  using time_type = Time;
59  //! \typedef strategy type
60  using dispersal_policy_type = DispersalPolicy;
61  //! \typedef value type to represent populations size
62  using N_value_type = typename DispersalPolicy::value_type;
63  //! \typedef time type
64  template<typename tree_type>
65  using forest_type = quetzal::coalescence::Forest<coord_type, tree_type>;
66  private:
67  using history_type = demography::History<coord_type, time_type, DispersalPolicy, StoragePolicy>;
68  history_type m_history;
69  public:
70  /**
71  * @brief Constructor

78  m_history(x_0, N_0, nb_generations)
79  {}
80  /**
81  * @brief Read-only access to the population size history.
82  * @detail Designed to be used for composing growth expression.
83  * @return a functor with signature `N_value_type fun(coord_type const& x,
84  * time_type const& t)` giving the population size in deme \f$x\f$ at time \f$t\f$.
85  */
86  auto pop_size_history() const noexcept
87  {
88  return std::cref(m_history.pop_sizes());
89  }
90  /**
91  * @brief Simulate the forward-in-time demographic expansion.
92  * \tparam Growth a growth functor, see param growth for details.
93  * \tparam Dispersal a dispersal kernel, see param kernel for details.
94  * @param growth a functor simulating \f$\tilde{N}_{x}^{t}\f$. The functor
95  * can possibly make use of pop_size_history. The signature
96  * of the function should be equivalent to the following:
97  * `unsigned int sim_growth(V &gen, const coord_type &x, const time_type &t);`
98  * @param kernel a functor representing the dispersal location kernel that
99  * simulates the coordinate of the next location conditionally
100  * to the current location \f$x\f$. The signature should be equivalent
101  * to `coord_type kernel(V &gen, const coord_type &x, const time_type &t);`
102  */
103  template<typename Generator, typename Growth, typename Dispersal>
104  void simulate_forward_demography(Growth growth, Dispersal kernel, Generator& gen)
105  {
106  m_history.simulate_forward(growth, kernel, gen);
107  }

134  /**
135  * @brief Coalesce a forest of nodes conditionally to the simulated demography.
136  *
137  */
138  template<typename Merger,typename Generator, typename F, typename Tree, typename U>
139  auto coalesce_along_spatial_history(forest_type<Tree> forest, F binary_op, Generator& gen, U make_tree) const
140  {
141  this->check_consistency(m_history, forest);
142  auto t = m_history.last_time();
143  while( (forest.nb_trees() > 1) && (t > m_history.first_time()) )
144  {
145  may_coalesce_colocated<Merger>(forest, t, gen, binary_op, make_tree);
146  migrate_backward(forest, t, gen);
147  --t;
148  }
149  may_coalesce_colocated<Merger>(forest, t, gen, binary_op, make_tree);
150  return forest;
151  }

265  template<typename T, typename U>
266  void check_consistency(T const& history, U const& forest) const
267  {
268  auto t = history.last_time();
269  for(auto const& it : forest.positions())
270  {
271  if( !history.pop_sizes().is_defined(it, t) || history.pop_sizes()(it, t) < forest.nb_trees(it))
272  {
273  throw std::domain_error("Simulated population size inferior to sampling size");
274  }
275  }
276  }

288  template<typename Generator, typename Forest>
289  void migrate_backward(Forest & forest, time_type t, Generator& gen) const noexcept
290  {
291  Forest new_forest;
292  for(auto const it : forest)
293  {
294  coord_type x = it.first;
295  new_forest.insert(m_history.backward_kernel(x, t, gen), it.second);
296  }
297  assert(forest.nb_trees() == new_forest.nb_trees());
298  forest = new_forest;
299  }

302  template<typename Merger, typename Generator, typename Tree, typename F, typename U>
303  void may_coalesce_colocated(forest_type<Tree>& forest, time_type const& t, Generator& gen, F binop, U make_tree) const noexcept
304  {
305  auto N = pop_size_history();
306  for(auto const & x : forest.positions())
307  {
308  auto range = forest.trees_at_same_position(x);
309  std::vector<Tree> v;
310  for(auto it = range.first; it != range.second; ++it)
311  {
312  v.push_back(it->second);
313  }
314  if(v.size() >= 2)
315  {
316  assert(N(x,t) >= 1 && "Population size should be positive for evaluating coalescence probability" );
317  auto last = Merger::merge(v.begin(), v.end(), N(x, t), make_tree(x,t), binop, gen );
318  forest.erase(x);
319  for(auto it = v.begin(); it != last; ++it)
320  {
321  forest.insert(x, *it);
322  }
323  }
324  }
325  }
