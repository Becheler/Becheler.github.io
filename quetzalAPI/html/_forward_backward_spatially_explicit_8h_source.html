128  template<typename Merger,typename Generator, typename F, typename Tree, typename U>
129  auto coalesce_along_spatial_history(forest_type<Tree> forest, F binary_op, Generator& gen, U make_tree) const
130  {
131  this->check_consistency(m_history, forest);
132  auto t = m_history.last_time();
133  while( (forest.nb_trees() > 1) && (t > m_history.first_time()) )
134  {
135  may_coalesce_colocated<Merger>(forest, t, gen, binary_op, make_tree);
136  migrate_backward(forest, t, gen);
137  --t;
138  }
139  may_coalesce_colocated<Merger>(forest, t, gen, binary_op, make_tree);
140  return forest;
141  }

255  template<typename T, typename U>
256  void check_consistency(T const& history, U const& forest) const
257  {
258  auto t = history.last_time();
259  for(auto const& it : forest.positions())
260  {
261  if( !history.pop_sizes().is_defined(it, t) || history.pop_sizes()(it, t) < forest.nb_trees(it))
262  {
263  throw std::domain_error("Simulated population size inferior to sampling size");
264  }
265  }

278  template<typename Generator, typename Forest>
279  void migrate_backward(Forest & forest, time_type t, Generator& gen) const noexcept
280  {
281  Forest new_forest;
282  for(auto const it : forest)
283  {
284  coord_type x = it.first;
285  new_forest.insert(m_history.backward_kernel(x, t, gen), it.second);
286  }
287  assert(forest.nb_trees() == new_forest.nb_trees());
288  forest = new_forest;
289  }

292  template<typename Merger, typename Generator, typename Tree, typename F, typename U>
293  void may_coalesce_colocated(forest_type<Tree>& forest, time_type const& t, Generator& gen, F binop, U make_tree) const noexcept
294  {
295  auto N = pop_size_history();
296  for(auto const & x : forest.positions())
297  {
298  auto range = forest.trees_at_same_position(x);
299  std::vector<Tree> v;
300  for(auto it = range.first; it != range.second; ++it)
301  {
302  v.push_back(it->second);
303  }
304  if(v.size() >= 2)
305  {
306  assert(N(x,t) >= 1 && "Population size should be positive for evaluating coalescence probability" );
307  auto last = Merger::merge(v.begin(), v.end(), N(x, t), make_tree(x,t), binop, gen );
308  forest.erase(x);
309  for(auto it = v.begin(); it != last; ++it)
310  {
311  forest.insert(x, *it);
312  }
313  }
314  }
315  }
