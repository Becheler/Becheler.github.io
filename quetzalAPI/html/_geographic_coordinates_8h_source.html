97  km distanceEarth(double lat1d, double lon1d, double lat2d, double lon2d) const {
98 
99  double earthRadiusKm = 6371.0;
100  double lat1r, lon1r, lat2r, lon2r, u, v;
101 
102  lat1r = deg2rad(lat1d);
103  lon1r = deg2rad(lon1d);
104  lat2r = deg2rad(lat2d);
105  lon2r = deg2rad(lon2d);
106 
107  u = sin((lat2r - lat1r)/2);
108  v = sin((lon2r - lon1r)/2);
109 
110  km d = 2.0 * earthRadiusKm * asin(sqrt(u * u + cos(lat1r) * cos(lat2r) * v * v));
111 
112  assert(d >= 0);
113 
114  return d;
115  }
