84  auto make_backward_distribution(coord_type const& x, time_type const& t) const
85  {
86 
87  std::vector<double> weights;
88  std::vector<coord_type> support;
89 
90  weights.reserve(m_flows->flux_to(x,t).size());
91  support.reserve(m_flows->flux_to(x,t).size());
92 
93  for(auto const& it : m_flows->flux_to(x,t) )
94  {
95  support.push_back(it.first);
96  weights.push_back(static_cast<double>(it.second));

265  {

272 
273  // TODO : optimize the definition_space function (for loop)
274  for(auto x : this->m_sizes->definition_space(t) )
275  {
276  auto N_tilde = sim_growth(gen, x, t);
277  landscape_individuals_count += N_tilde;
278  if(N_tilde >= 1)
279  {
280  for(unsigned int ind = 1; ind <= N_tilde; ++ind)

282  auto y = kernel(gen, x, t);

286  }
