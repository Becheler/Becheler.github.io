58  /**
59  * @brief Expands the demographic database.
60  *
61  * @tparam T a functor for representing the growth process.
62  * @tparam U a functor for representing the dispersal process.
63  * @tparam V a random number generator.
64  *
65  * @param nb_generations the number of generations to simulate
66  * @param sim_growth a functor simulating \f$\tilde{N}_{x}^{t}\f$. The functor
67  * can possibly internally use a reference on the population
68  * sizes to represent the time dependency. The signature
69  * of the function should be equivalent to the following:
70  * `unsigned int sim_growth(V &gen, const coord_type &x, const time_type &t);`
71  * @param kernel a functor representing the dispersal location kernel that
72  * simulates the coordinate of the next location conditionally
73  * to the current location \f$x\f$. The signature should be equivalent
74  * to `coord_type kernel(V &gen, const coord_type &x, const time_type &t);`
75  *
76  * @exception std::domain_error if the population goes extincted before the simulation is completed.
77  *
78  * \section Example
79  * \snippet demography/test/History/Individual_based_history_test.cpp Example
80  *
81  * \section Output
82  * \include demography/test/History/Individual_based_history_test.output
83  */
84  template<typename T, typename U, typename V>
85  void expand(unsigned int nb_generations, T sim_growth, U kernel, V& gen)
86  {
87  for(unsigned int g = 0; g < nb_generations; ++g)
88  {
89  auto t = this->last_time();
90  auto t_next = t; ++ t_next;
91 
92  this->m_times.push_back(t_next);
93 
94  unsigned int landscape_individuals_count = 0;
95 
96  // TODO : optimize the definition_space function (for loop)
97  for(auto x : this->m_sizes->definition_space(t) )
98  {
99  auto N_tilde = sim_growth(gen, x, t);
100  landscape_individuals_count += N_tilde;
101  if(N_tilde >= 1)
102  {
103  for(unsigned int ind = 1; ind <= N_tilde; ++ind)
104  {
105  coord_type y = kernel(gen, x);
106  this->m_flows->add_to_flux_from_to(x, y, t, 1);
107  this->m_sizes->operator()(y, t_next) += 1;
108  }
109  }
110  }
111 
112  if(landscape_individuals_count == 0)
113  {
114  throw std::domain_error("Landscape populations went extinct before sampling");
115  }
116 
117  }
118  }

144  *
145  * @tparam T a functor for representing the growth process.
146  * @tparam U a functor for representing the dispersal process.
147  * @tparam V a random number generator.
148  *
149  * @param nb_generations the number of generations to simulate.
150  * @param sim_growth a functor simulating \f$\tilde{N}_{x}^{t}\f$. The functor
151  * can possibly internally use a reference on the population sizes database
152  * to represent the time dependency. The signature of the function should
153  * be equivalent to the following:
154  * `unsigned int sim_growth(V &gen, const coord_type &x, const time_type &t);`.
155  * @param kernel a functor representing the dispersal transition matrix.
156  * The signature of the function should be equivalent to
157  * `double kernel( const coord_type & x, const coord_type &y);`
158  * and the function should return the probability for an individual
159  * to disperse from \f$x\f$ to \f$y\f$ at time \f$t\f$.
160  * The expression `kernel.arrival_state()` must be valid
161  * and returns an iterable container of geographic coordinates
162  * indicating the transition kernel state space.
163  *
164  * @section Example
165  * @snippet demography/test/History/Mass_based_history_test.cpp Example
166  *
167  * @section Output
168  * @include demography/test/History/Mass_based_history_test.output
169  */
170  template<typename T, typename U, typename V>
171  void expand(unsigned int nb_generations, T sim_growth, U kernel, V& gen)
172  {
173  for(unsigned int g = 0; g < nb_generations; ++g)
174  {
175  auto t = this->last_time();
176  auto t_next = t; ++ t_next;
177 
178  this->m_times.push_back(t_next);
179 
180  unsigned int landscape_individuals_count = 0;
181 
182  for(auto x : this->m_sizes->definition_space(t) )
183  {
184  auto N_tilde = sim_growth(gen, x, t);
185 
186  for(auto const& y : kernel.arrival_space(x) )
187  {
188  auto m = kernel(x, y);
189  assert(m >= 0.0 && m <= 1.0);
190 
191  // TODO : I changed floor to ceil for oversea migration
192  double nb_migrants = std::ceil(m * static_cast<double>(N_tilde));
193  if(nb_migrants >= 0){
194  landscape_individuals_count += nb_migrants;
195  this->m_flows->set_flux_from_to(x, y, t, nb_migrants);
196  this->m_sizes->operator()(y, t_next) += nb_migrants;
197  }
198  }
199  }
200 
201  if(landscape_individuals_count == 0)
202  {
203  throw std::domain_error("Landscape populations went extinct before sampling");
204  }

206  }
