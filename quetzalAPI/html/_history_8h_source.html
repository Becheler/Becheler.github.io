142  template<typename Generator>
143  coord_type backward_kernel(coord_type const& x, time_type t, Generator& gen) const
144  {
145  --t;
146  assert(m_flows->flux_to_is_defined(x,t));
147  if( ! m_kernel->has_distribution(x, t))
148  {
149  m_kernel->set(x, t, make_backward_distribution(x, t));
150  }
151  return m_kernel->operator()(gen, x, t);
152  }
153

166  auto make_backward_distribution(coord_type const& x, time_type const& t) const
167  {
168 
169  std::vector<double> weights;
170  std::vector<coord_type> support;
171 
172  weights.reserve(m_flows->flux_to(x,t).size());
173  support.reserve(m_flows->flux_to(x,t).size());
174 
175  for(auto const& it : m_flows->flux_to(x,t) )
176  {
177  support.push_back(it.first);
178  weights.push_back(static_cast<double>(it.second));
179  }
180 
181  return discrete_distribution_type(std::move(support),std::move(weights));
182  }

251  {
252  for(unsigned int g = 0; g < nb_generations; ++g)
253  {
254  auto t = this->last_time();
255  auto t_next = t; ++ t_next;
256 
257  this->m_times.push_back(t_next);
258 
259  unsigned int landscape_individuals_count = 0;
260 
261  // TODO : optimize the definition_space function (for loop)
262  for(auto x : this->m_sizes->definition_space(t) )
263  {
264  auto N_tilde = sim_growth(gen, x, t);
265  landscape_individuals_count += N_tilde;
266  if(N_tilde >= 1)
267  {
268  for(unsigned int ind = 1; ind <= N_tilde; ++ind)
269  {
270  coord_type y = kernel(gen, x);
271  this->m_flows->add_to_flux_from_to(x, y, t, 1);
272  this->m_sizes->operator()(y, t_next) += 1;
273  }
274  }
275  }
276 
277  if(landscape_individuals_count == 0)
278  {
279  throw std::domain_error("Landscape populations went extinct before sampling");
280  }

282  }
283  }
