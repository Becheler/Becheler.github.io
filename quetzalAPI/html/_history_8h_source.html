299  {

302 
303  this->m_times.push_back(t_next);
304 
305  unsigned int landscape_individuals_count = 0;
306 
307  // TODO : optimize the definition_space function (for loop)
308  for(auto x : this->m_sizes->definition_space(t) )
309  {
310  auto N_tilde = sim_growth(gen, x, t);
311  landscape_individuals_count += N_tilde;
312  if(N_tilde >= 1)
313  {
314  for(unsigned int ind = 1; ind <= N_tilde; ++ind)
315  {
316  auto y = kernel(gen, x, t);
317  this->m_flows->add_to_flux_from_to(x, y, t, 1);
318  this->m_sizes->operator()(y, t_next) += 1;
319  }
320  }
321  }
322 
323  if(landscape_individuals_count == 0)
324  {
325  throw std::domain_error("Landscape populations went extinct before sampling");
