329  template<typename Growth, typename Dispersal, typename Generator>
330  void expand(unsigned int nb_generations, Growth sim_growth, Dispersal kernel, Generator& gen)
331  {
332  for(unsigned int g = 0; g < nb_generations; ++g)
333  {
334  auto t = this->last_time();
335  auto t_next = t; ++ t_next;
336 
337  this->m_times.push_back(t_next);
338 
339  unsigned int landscape_individuals_count = 0;
340 
341  // TODO : optimize the definition_space function (for loop)
342  for(auto x : this->m_sizes->definition_space(t) )
343  {
344  auto N_tilde = sim_growth(gen, x, t);
345  landscape_individuals_count += N_tilde;
346  if(N_tilde >= 1)
347  {
348  for(unsigned int ind = 1; ind <= N_tilde; ++ind)
349  {
350  auto y = kernel(gen, x, t);
351  this->m_flows->add_to_flux_from_to(x, y, t, 1);
352  this->m_sizes->operator()(y, t_next) += 1;
353  }
354  }
355  }
356 
357  if(landscape_individuals_count == 0)
358  {
359  throw std::domain_error("Landscape populations went extinct before sampling");
360  }
361 
362  }
363  }
