172  {
173 
174  std::vector<double> weights;

177  weights.reserve(m_flows->flux_to(x,t).size());
178  support.reserve(m_flows->flux_to(x,t).size());

350  template<typename Growth, typename Dispersal, typename Generator>
351  void expand(unsigned int nb_generations, Growth sim_growth, Dispersal kernel, Generator& gen)
352  {
353  for(unsigned int g = 0; g < nb_generations; ++g)
354  {
355  auto t = this->last_time();
356  auto t_next = t; ++ t_next;
357 
358  this->m_times.push_back(t_next);
359 
360  unsigned int landscape_individuals_count = 0;
361 
362  // TODO : optimize the definition_space function (for loop)
363  for(auto x : this->m_sizes->definition_space(t) )
364  {
365  auto N_tilde = sim_growth(gen, x, t);
366  landscape_individuals_count += N_tilde;
367  if(N_tilde >= 1)
368  {
369  for(unsigned int ind = 1; ind <= N_tilde; ++ind)
370  {
371  auto y = kernel(gen, x, t);
372  this->m_flows->add_to_flux_from_to(x, y, t, 1);
373  this->m_sizes->operator()(y, t_next) += 1;
374  }
375  }
376  }
377 
378  if(landscape_individuals_count == 0)
379  {
380  throw std::domain_error("Landscape populations went extinct before sampling");
381  }
382 
383  }
384  }

474

476 
477  for(auto x : this->m_sizes->definition_space(t) )
