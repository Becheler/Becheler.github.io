36  /*!
37  * \ingroup coal_spectrum
38  * \brief Generates random occupancy spectrum that are distributed according
39  * to the associated probability function given by von Mises (1939).
40  * The entropy is acquired by calling g.operator().
41  * \remark The probability of an occupancy spectrum is described in the book
42  * "Urn models and their application: an approach
43  * to modern discrete probability theory" (by John and Kotz, 1977, p.115).
44  * \tparam UnaryPredicate Function object class acting as a predicate, taking
45  * the probability of the spectrum as argument and returning a boolean.
46  * The signature should be `bool UnaryPredicate::operator()(double)`
47  * The number of possible configurations grows very fast with n and m,
48  * so memorizing all of them quickly becomes intractable.
49  * As most of the configurations have extremely low output probability,
50  * you can choose to filter the spectrum according to their sampling
51  * probability, keeping only those which probability is superior to a
52  * given threshold (or other criterion). If the predicate is evaluated
53  * to true, the spectrum and its probability will be saved in the
54  * distribution. If the predicate is evaluated to false, the spectrum
55  * and its probability will be discarded. By default, ReturnAlwaysTrue
56  * policy is used, so all spectra are kept.
57  * \tparam SpectrumHandler Function object class modifying the spectrum to optimize speed and
58  * memory use. The signature should be
59  * `OccupancySpectrum SpectrumHandler::operator()(OccupancySpectrum &&)`.
60  * By default the Identity policy wil not modify the spectra.
61  * The RetrieveLastEmptyUrns will retrieve all the last urns that are
62  * empty, making later iteration over a spectrum faster.
63  * \tparam Int A big integer class for probability computation
64  * \tparam FLoat A big float class for probability computation
65  * \section Example
66  * \snippet coalescence/occupancy_spectrum/test/OccupancySpectrumDistribution/OccupancySpectrumDistribution_test.cpp Example
67  * \section Output
68  * \include coalescence/occupancy_spectrum/test/OccupancySpectrumDistribution/OccupancySpectrumDistribution_test.output
69  */

232  auto make_callback(unsigned int k, unsigned int N){
233  return [this, k, N](spectrum_type&& M_j){
234  auto p = compute_probability(k, N, M_j);
235  if(this->m_pred(p)){
236  this->m_support.push_back(SpectrumHandler::handle(std::move(M_j)));
237  this->m_probas.push_back(p);
238  }
239  };

242  static Int Factorial(int num) {
243  Int fact = 1;
244  for(int i=num; i>1; --i){
245  fact *= i;
246  }
247  return fact;
248  }

250 
251  static double compute_probability(unsigned int n, unsigned int m, spectrum_type const& m_js) {
252  Float numerator(Factorial(m)*Factorial(n));
253  Float denominator(1.0);
254  int j = 0;
255  for(auto m_j : m_js){
256  denominator *= boost::multiprecision::pow(Float(Factorial(j)),Float(m_j)) * Float(Factorial(m_j));
257  ++j;
258  }
259  denominator *= Float(std::pow(m,n));
260  Float result = numerator/denominator;
261  return static_cast<double>(result) ;
262  }
