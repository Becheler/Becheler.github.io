159  static double compute_probability(unsigned int n, unsigned int m, spectrum_type const& m_js) {
160  Float numerator(Factorial(m)*Factorial(n));
161  Float denominator(1.0);
162  int j = 0;
163  for(auto m_j : m_js){
164  denominator *= boost::multiprecision::pow(Float(Factorial(j)),Float(m_j)) * Float(Factorial(m_j));
165  ++j;
166  }
167  denominator *= Float(std::pow(m,n));
168  Float result = numerator/denominator;
169  return static_cast<double>(result) ;
