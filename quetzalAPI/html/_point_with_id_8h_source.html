17  /// @brief Find the index of a point in a vector
18  /// @param p the point to find
19  /// @param vect the vector in points in which to search for the point index
20  /// @return the index of the point in the vector
21  ///
22  template<typename Point>
23  size_t getIndexOfPointInVector(Point const& p, std::vector<Point> const& vect)
24  {
25  auto it = std::find(vect.begin(), vect.end(), p);
26  assert(it != vect.end());
27  return std::distance(vect.begin(), it);
28  }
29  ///
30  /// @brief Class to associate an indexing semantics to a collection of points
31  ///
32  /// @details it is mainly used when geographic coordinates are stored in a vector
33  /// and that a pairwise matrix has to be computed: PointWithId class make
34  /// easy to switch between geographic and indexing semantics.
35  /// @tparam Point the class for which an indexing semantics is needed, usually GeographicCoordinates
36  ///
37  template<typename Point>
38  class PointWithId
39  {
40  public:
41  /*!
42  \brief Constructor
43  \param vect reference to the vector of coordinates
44  \param index index of the point in the vector
45  */
46  PointWithId(std::vector<Point> const& vect, size_t index) :
47  m_vector{vect}, m_index{index}
48  {}
49  /*!
50  \brief Constructor

55  PointWithId(std::vector<Point> const& vect, Point const& p) :

58  /*!
