62 // String extractions, dependant on file structure
63  void analyze(std::string const& path) {
64  std::ifstream file(path);
65  std::string line;
66  unsigned int current_line = 0;
67 
68  while ( std::getline( file, line ) ){
69  if(current_line == 0){
70  m_loci_names = read_loci_names(erase_coordinates(line));
71  }else{
72  auto x = read_coordinates(line);
73  auto ind = read_individual(erase_coordinates(line));
74  m_sample.add(x, ind);
75  }
76  ++current_line;
77  }

99  coord_type read_coordinates(std::string const& line) const {
100  double lat = 0.;
101  double lon = 0.;
102  std::istringstream iss( line );
103  string elem;
104  unsigned int pos = 0;
105  while ( std::getline( iss, elem, ' ' ) && pos <= 1 ) {
106  if(pos == 0){
107  lat = std::stod(elem);
108  }else if( pos == 1){
109  lon = std::stod(elem);
110  }
111  ++pos;
112  }
113  coord_type x(lat, lon);
114  return x;
115  }

146  }
