QUETZAL
The Coalescence Template Library
template<class CellT >
CellT & quetzal::coalescence::Tree< CellT >::cell ( )

Cell accessor.

Returns
a reference to the tree cell
Remarks
Read-and-write data access

Example

#include <string>
#include "assert.h"
#include <iostream>
int main(){
using quetzal::coalescence::Tree;
using std::string;
/* Topology :
* a
* / \
* b c
*/
Tree<string> a("a");
assert( ! a.has_parent() );
assert( ! a.has_children() );
auto& b = a.add_child("b");
assert( b.has_parent() );
assert( a.has_children() );
auto& c = a.add_child("c");
assert(c.cell() == "c");
c.cell() = "d";
assert(c.cell() != "c");
std::cout << "done" << std::endl;
return 0;
}
