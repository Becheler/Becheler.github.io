template<typename Space , typename Time , typename Value >
Value & quetzal::demography::PopulationSize< Space, Time, Value >::operator() ( coord_type &&  x,
time_type &&  t 
)

Population size at deme x at time t.

Returns
a reference to the value, initialized with value_type default constructor
Remarks
operator allows for more expressive mathematical style in client code

Example

#include <iostream> // std::cout
#include <algorithm> // std::copy
#include <iterator> // std::ostream_iterator
#include <string>
#include "assert.h"
int main(){
using coord_type = std::string;
using time_type = unsigned int;
using size_type = int;
time_type t0 = 2017;
coord_type x0 = "Paris";
size_type N0 = 12;
quetzal::demography::PopulationSize<coord_type, time_type, size_type> N;
assert(N.is_defined(x0, t0) == false );
N(x0,t0) = N0;
assert(N.is_defined(x0, t0) );
assert( N.get(x0,t0) == N0 );
assert( N.definition_space(t0).size() == 1 );
N.set("Bordeaux", t0, 2*N0);
auto X = N.definition_space(t0);
std::copy(X.begin(), X.end(), std::ostream_iterator<coord_type>(std::cout, " "));
return 0;
}

Output

Paris Bordeaux
