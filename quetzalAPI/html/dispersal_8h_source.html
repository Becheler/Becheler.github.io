25 /* Class to represent the distance matrix between each pair of points in a set.
26  *
27  * @details Useful e.g. to implement distance based dispersal kernels.
28  * @tparam Space the type of points
29  * @tparam Value the type of the distance value between pairs of points
30  */
31 template<typename Space, typename Value>
32 class SymmetricDistanceMatrix
33 {
34 
35 public:
36 
38  template<typename T>
39  using matrix_type = boost::numeric::ublas::symmetric_matrix<T>;
40 
42  using coord_type = Space;
43 
45  using value_type = Value;
46 
47  /* Construct the matrix by computing distance between each pair of points in the set.
48  *
49  * @tparam F the distance function, with signature equivalent to the following:
50  * `value_type f(const coord_type & x, const coord_type & y)`
51  */
52  template<typename F>
53  SymmetricDistanceMatrix(std::vector<coord_type> const& points, F f):
54  m_points(points),
55  m_matrix(construct_matrix(points, f))
56  {}
57 
59  friend std::ostream& operator <<(std::ostream& Stream, const SymmetricDistanceMatrix & M)
60  {
61  Stream << M.m_matrix;
62  return Stream;
63  }
64 
65  /* Apply a functor over the matrix and return the resulting new matrix
66  *
67  * @tparam UnaryOperation the transformation to apply to each matrix value.
68  * @param op UnaryOperation object. The signature should be equivalent to
69  * `Ret op(value_type)`, with `Ret` being possibly any Copiable type.
70  */
71  template<typename UnaryOperation>
72  auto apply(UnaryOperation op) const
73  {
74  using other_type = typename std::result_of_t<UnaryOperation(std::remove_cv_t<std::remove_reference_t<const value_type &>>)>;
75  matrix_type<other_type> m (m_points.size(), m_points.size());
76  for (unsigned i = 0; i < m.size1 (); ++ i){
77  for (unsigned j = 0; j <= i; ++ j){
78  m (i, j) = op(m_matrix(i, j));
79  }
80  }
81  return SymmetricDistanceMatrix<coord_type, other_type>(m_points, m);
82  }
83 
85  std::vector<coord_type> const& points(){return m_points;}
86 
87  /* For a given point, give the distance to all other points
88  * @param x the given point
89  * @return a boost::numeric::ublas::row
90  */
91  auto at(coord_type const& x) const
92  {
93  return boost::numeric::ublas::row(m_matrix, ID(x));
94  }
95 
96  value_type operator()(coord_type const& x, coord_type const& y) const {
97  return m_matrix(ID(x),ID(y));
98  }
99 
100 private:
101 
102  std::vector<Space> m_points;
103  matrix_type<Value> m_matrix;
104 
105  unsigned int ID(coord_type const& x) const {
106  auto it = std::find(m_points.begin(), m_points.end(), x);
107  assert( it != m_points.end());
108  return std::distance(m_points.begin(), it);
109  }
110 
111  template<typename F>
112  auto construct_matrix(std::vector<coord_type> const& points, F f) const
113  {
114  matrix_type<value_type> m (points.size(), points.size());
115  for (unsigned i = 0; i < m.size1 (); ++ i)
116  {
117  for (unsigned j = 0; j <= i; ++ j)
118  {
119  m (i, j) = f(points.at(i), points.at(j));
120  }
121  }
122  return m;
123  }
124 
125 };

127 // @brief Constructs a symmetric matrix of distances between points of the space.
128 // @tparam Space Demes identifiers (geographic points)
129 // @tparam F a functor implementing a symetric distance relationship between two
130 // geographic points.
131 // @param points the set of geographic points to consider
132 // @param f a functor giving the distance between two demes and returning any copiable type T.
133 // The signature should be equivalent to the following `T f(Space const& a, Space const& b)`
134 // @return a symmetric distance matrix
135 template<typename Space, typename F>
136 auto make_symmetric_distance_matrix(std::vector<Space> const& points, F f){
137  using value_type = typename std::result_of_t<F(std::remove_cv_t<std::remove_reference_t<const Space &>>,
138  std::remove_cv_t<std::remove_reference_t<const Space &>>)>;
139  return SymmetricDistanceMatrix<Space, value_type>(points, f);
140 }

143 // @brief Sampling distribution in the demic structure
144 // @tparam Space the deme identifier type (may be integers, geographic coordinates...)
145 //
146 // Allows to sample a destination point conditioning on the present location
147 template<typename Space>
148 class DiscreteLocationSampler
149 {
150 public:
151 
152  // @typedef the deme identifier
153  using coord_type = Space;
154 
155 private:
156  std::vector<coord_type> m_points;
157 
158  using ID_type = unsigned int;
159  std::unordered_map<coord_type, ID_type> m_dico;
160 
161  using distribution_type = std::discrete_distribution<ID_type>;
162  mutable std::vector<distribution_type> m_kernels;
163 
164  std::unordered_map<coord_type, unsigned int> make_dico(std::vector<coord_type> const& points) const
165  {
166  unsigned int position = 0;
167  std::unordered_map<coord_type, unsigned int> m;
168  for(auto const& it : points)
169  {
170  m.emplace(it, position);
171  ++position;
172  }
173  return m;
174  }
175 
176 public:
177 
178  DiscreteLocationSampler(std::vector<coord_type> const& points, std::vector<distribution_type> && kernels):
179  m_points(points),
180  m_dico(make_dico(points)),
181  m_kernels(kernels)
182  {}
183 
184  // @brief samples a destination in the location probability distribution
185  // @tparam Generator a random number generator
186  // @param x the present location
187  // @param gen a random number generator
188  // @return a deme sampled in the discrete probability distribution of arrivals.
189  template<typename Generator>
190  coord_type operator()(coord_type const& x, Generator & gen) const
191  {
192  auto id = m_dico.at(x);
193  auto & dist = m_kernels.at(id);
194  return m_points.at(dist(gen));
195  }
196 
197 };

199 // @brief Constructs a discrete location kernel
200 // @tparam M a distance matrix
201 // @tparam Kernel a dispersal kernel
202 // @return a DiscreteLocationSampler
203 //
204 template<typename Kernel, typename M>
205 auto make_discrete_location_sampler(M const& d, typename Kernel::param_type const& p)
206 {
207  auto f = [p](auto r){return Kernel::pdf(r, p);};
208  auto weights = d.apply(f);
209  std::vector<std::discrete_distribution<unsigned int>> discrete_kernels;
210  discrete_kernels.reserve(weights.points().size());
211  for(auto const& x : weights.points())
212  {
213  auto x_weights = weights.at(x);
214  discrete_kernels.push_back(std::discrete_distribution<unsigned int>(x_weights.begin(), x_weights.end()));
215  }
216  return DiscreteLocationSampler<typename M::coord_type>(weights.points(), std::move(discrete_kernels));
217 }
