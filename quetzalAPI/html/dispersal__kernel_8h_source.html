227  static double pdf(double r, param_type const& param)
228  {
229  double a1 = param.a1();
230  double a2 = param.a2();
231  double p = param.p();
232  assert(a1 > 0 && a2 > 0 && p > 0 && p < 1 && r >= 0);
233  return p/(M_PI*a1*a1) * std::exp(-(r*r)/(a1*a1)) + (1-p)/(M_PI*a2*a2) * std::exp(-(r*r)/(a2*a2)) ;
