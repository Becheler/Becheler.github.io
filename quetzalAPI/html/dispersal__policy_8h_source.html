283  auto arrival_space(coord_type const& x) const
284  {
285  auto it = m_arrival_space_cash.find(x);
286  if(it != m_arrival_space_cash.end())
287  {
288  return it->second;
289  }
290  else
291  {
292  auto p = m_arrival_space_cash.emplace(x, retrieve_non_zero_arrival_space(x));
293  return p.first->second;
294  }
295  }
299  auto retrieve_non_zero_arrival_space(coord_type const& x) const
300  {
301  using it1_t = typename matrix_type::const_iterator1;
302  using it2_t = typename matrix_type::const_iterator2;
303  auto i = getIndexOfPointInVector(x, m_coords);
304  it1_t it1 = m_matrix.begin1();
305  std::advance(it1, i);
306  std::vector<coord_type> v;
307  v.reserve(m_matrix.size2()); // worst case
308  for (it2_t it2 = it1.begin(); it2 != it1.end(); it2++)
309  {
310  if(*it2 > 0.0) v.push_back(m_coords.at(it2.index2()));
311  }
312  return v;
313  }

318  {
319  return m_matrix(getIndexOfPointInVector(x, m_coords), getIndexOfPointInVector(y, m_coords) );

443  std::vector<coord_type> arrival_space(coord_type const& x) const
444  {
445  auto v = m_get_neighbors(x);
446  // let's add focal deme to have all possible arrival demes
447  v.push_back(x);
448  return v;
449  }
450 
454  auto operator()(coord_type const& from, coord_type const& to)
455  {
456  if(from == to)
457  {
458  return 1.0 - m_emigrant_rate;
459  }
460  else
461  {
462  return m_emigrant_rate * friction_weights(from, to);
463  }
464  }
465  private:
466  double friction_weights(coord_type const& x, coord_type const& y) const
467  {
468  // TODO does it copy the whole vector?
469  auto neighbors = m_get_neighbors(x);
470  std::vector<double> v(neighbors.size(), 0.0);
471  std::transform(neighbors.begin(), neighbors.end(), v.begin(), [this](coord_type const& z) { return 1.0/(this->m_friction(z));} );
472  double sum = std::accumulate(v.begin(), v.end(), 0.0);
473  return 1.0/(m_friction(y)*sum);
474  }
475  }; // end neighboring_migration
