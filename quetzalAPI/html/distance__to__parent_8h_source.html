54  //! Treatment to operate on a DFS on the tree to compute branches length
55  struct treatment {
56  auto operator()(tree_type & node) const {
57  if(node.has_parent()){
58  int d = node.cell().time() - node.parent().cell().time();
59  int l = std::abs(d);

62  node.cell().distance_to_parent(0);
63  }
64  }

91  template<typename Generator>
92  tree_type find_mrca(forest_type const& forest, time_type const& first_time, Generator& gen) const {
93 
94  time_type t_curr = first_time;
95 
96  auto WF_init = [&t_curr](time_type const& t){
97  t_curr -= t;
98  return tree_type(t_curr);
99  };
100 
101  using WF_model = quetzal::DiscreteTimeWrightFisher;
102  auto tree = WF_model::coalesce(forest, m_ancestral_Wright_Fisher_N, gen, branch(), WF_init);
103  treatment computer;
104  tree.visit_subtrees_by_pre_order_DFS(computer);
105  return tree;
106  }
