152  template<typename Generator>
153  tree_type find_mrca(forest_type const& forest, time_type const& first_time, Generator& gen) const
154  {
155  time_type t_curr = first_time;
156  auto WF_init = [&t_curr](time_type const& t)
157  {
158  t_curr -= t;
159  return tree_type(t_curr);
160  };
161  using WF_model = quetzal::DiscreteTimeWrightFisher;
162  auto tree = WF_model::coalesce(forest, m_ancestral_Wright_Fisher_N, gen, branch(), WF_init);
163  treatment computer;
164  tree.visit_subtrees_by_pre_order_DFS(computer);
165  return tree;
166  }
