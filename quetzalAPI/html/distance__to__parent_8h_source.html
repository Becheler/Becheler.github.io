92  tree_type find_mrca(forest_type const& forest, time_type const& first_time, Generator& gen) const {
93 
94  time_type t_curr = first_time;
95 
96  auto WF_init = [&t_curr](time_type const& t){
97  t_curr -= t;
98  return tree_type(t_curr);
99  };
100 
101  using WF_model = quetzal::DiscreteTimeWrightFisher;
102  auto tree = WF_model::coalesce(forest, m_ancestral_Wright_Fisher_N, gen, branch(), WF_init);
103  treatment computer;
104  tree.visit_subtrees_by_pre_order_DFS(computer);

256  /*
257  * @brief Create a forest from a vector of individuals, extracting their location, name and sampling time.
258  *
259  * @tparam T type of an individual
260  * @tparam F1 a unary operation with signature equivalent to 'coord_type fun(const T& i)'
261  * @tparam F2 a unary operation with signature equivalent to 'std::string fun(const T& i)'
262  *
263  * @param sample a collection of individuals (gene copies)
264  * @param sampling_time the time of sampling
265  * @param get_location a unary operation taking a sampled individual of type T as an argument and returning its geographic location
266  * @param get_name a functor taking a sampled individual of type T as an argument and returning a std::string
267  */
