97 /*
98 support base class for functors delagating their operator() to other functor(s).
99 */
100 template <typename... Functors>
101 struct composite_functor {
102  using functors_type = std::tuple<Functors...>;
103 
104  template <std::size_t I>
105  using return_type = return_type< typename std::tuple_element<I, functors_type>::type >;
106 
107  template <std::size_t I>
108  static constexpr std::size_t arguments_count() {return Callable_traits<typename std::tuple_element<I, functors_type>::type>::arguments_count;}
109 
110  using arguments_type = typename std::common_type< typename Callable_traits<Functors>::arguments_type... >::type;
111 
112  constexpr composite_functor(Functors const&... functors) : functors(functors...) {
113  // all Functors shall have the same call arguments (which will be those of the resulting functor)
114  static_assert( meta::all< functor_has_interface<Functors, arguments_type>::value... >(),
115  "incompatible functors (arguments differs)"
116  );
117  }
118 
119  //callability check
120  template<typename... Args>
121  static constexpr bool call_check() {
122  //shall we check against interface or against each functors
123  return meta::all< callable_with<Functors, Args...>::value... >();
124  }
125 
126  template <std::size_t I>
127  constexpr auto const& f() const {return std::get<I>(functors);}
128 
129  functors_type functors;
130 };

391 template <typename Outer, typename... Inners>
392 struct compose_t: public composite_functor<Inners...> {
393  using base_t = composite_functor<Inners...>;
394  using value_type = return_type<Outer>;
395 
396  constexpr
397  compose_t(Outer outer, Inners... inners) : base_t(inners...), outer(outer) {
398  // there must be exactly as many tied functors as Outer arguments
399  static_assert(
400  sizeof...(Inners) == arguments_count<Outer>(),
401  "wrong functors count"
402  );
403 
404  // each tied functor must return a value compatible with the matching Outer's argument
405  static_assert(
406  composable_check(std::index_sequence_for<Inners...>{}),
407  "uncomposable functors"
408  );
409  }
410 
411  template <typename... Args>
412  constexpr
413  value_type operator()(Args... args) const {
414  static_assert(base_t::template call_check<Args...>(), "wrong arguments type or count");
415 
416  return impl( std::index_sequence_for<Inners...>{} , args...);
417  }
418 
419  // *** call details *** //
420  //actual call
421  template<std::size_t... Is, typename... Args>
422  constexpr
423  value_type impl(std::index_sequence<Is...>, Args... args) const {
424  return outer( base_t::template f<Is>()(args...)... );
425  }
426 
427  // *** build details *** //
428  template<std::size_t... Is>
429  static constexpr
430  bool composable_check(std::index_sequence<Is...>){
431  return meta::all<std::is_convertible< return_type<Inners>, argument_type<Outer, Is> >::value...>();
432  }
433 
434  Outer outer;
435 };
