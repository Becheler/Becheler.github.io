119 
120  // directly go to binary merge.
121  auto m_it = sp.cbegin();
122  std::advance(m_it, 2);
123  int j = 2;
124 
125  while(m_it != sp.cend()){
126  for(unsigned int i = 1; i <= *m_it; ++i){ // loop on the m_j parents
127  *first = op(init, *first);
128  for(int k = 1; k < j; ++k){ // loop on the others children
129  *first = op(*first, *(--last));
130  }
131  ++first;
132  }
133  ++j;
134  ++m_it;
135  }
136  return last;
