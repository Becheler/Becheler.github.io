125 
126  // directly go to binary merge.
127  auto m_it = sp.cbegin();
128  std::advance(m_it, 2);
129  int j = 2;

131  while(m_it != sp.cend()){
132  for(unsigned int i = 1; i <= *m_it; ++i){ // loop on the m_j parents
133  if constexpr (std::is_invocable<T>::value){
134  *first = op(init(), *first);

139  *first = op(*first, *(--last));

142  }
143  ++j;

145  }
