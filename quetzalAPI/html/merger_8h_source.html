36  /**
37  * \brief merges 2 randomly selected elements in a range.
38  *
39  * With a probability \f$1/N\f$, merges 2 elements selected uniformely at random in a range.
40  *
41  * \param first iterator at the begin of the range
42  * \param last iterator to the past-the-end element.
43  * \param N the number of individuals in the population
44  * \param init the value at which parent is initialized
45  * \param op binary operation function object that will be applied for branching a
46  child to its parent. The binary operator takes the parent value a
47  (initialized to init) and the value of the child b. The signature
48  of the function should be equivalent to the following:
49  `Ret fun(const Type1 &parent, const Type2 &child);`
50  * \param g a random generator that must meet the requirements of UniformRandomBitGenerator concept.
51  * \return An iterator to the element that follows the last element of the nodes remaining after coalescence.
52  The function cannot alter the properties of the object containing the range of elements
53  (i.e., it cannot alter the size of an array or a container):
54  signaling the new size of the shortened range is done by returning an iterator to the element
55  that should be considered its new past-the-end element. The range between first and
56  this iterator includes all the remaining nodes in the sequence.
57  *
58  * \remark BidirectionalIterator must meet the requirements of ValueSwappable and RandomAccessIterator.
59  */
60  template<class BidirectionalIterator, class T, class BinaryOperation, class Generator>
61  static auto
62  merge(BidirectionalIterator first, BidirectionalIterator last, unsigned int N, T init, BinaryOperation const& binop, Generator& g)
63  {
64  assert(N >= 1 && "Population size should be more than 1 for evaluating coalescence probability" );
65  assert(std::distance(first, last) > 1 && "Coalescence should operate on a range containing more than one element.");
66  double k = static_cast<double>(std::distance(first, last));
67  double coal_proba = (k*(k-1.0))/static_cast<double>(2*N);
68  std::bernoulli_distribution d(coal_proba);
69  if( d(g) ){
70  last = binary_merge(first, last, init, binop, g);
71  }
72  return last;
73  }

95  */
96  template<class BidirectionalIterator, class Generator>

98  merge(BidirectionalIterator first, BidirectionalIterator last, unsigned int N, Generator& g)
99  {
100  assert(N >= 1 && "Population size should be positive for evaluating coalescence probability" );
101  assert(std::distance(first, last) > 1 && "Coalescence should operate on a range containing more than one element.");
102  using T = typename BidirectionalIterator::value_type;
103  return merge(first, last, N, T(), std::plus<T>(), g);
104  }
