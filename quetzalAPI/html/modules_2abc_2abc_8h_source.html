159  template<class ObservedDataType, typename Generator>
160  std::vector<param_type> rubin_rejection_sampler(unsigned int N, ObservedDataType const& y, Generator& g) const {
161  std::vector<param_type> sample;
162  sample.reserve(N);
163  while(sample.size() != N)
164  {
165  auto params = param_type(priors(g));
166  if( y == generative_model(g, params))

168  sample.push_back(params);

170  }

280  template<class StatisticsType, typename Generator>
281  auto sample_prior_predictive_distribution(unsigned int N, StatisticsType const& eta, Generator& g) const {
282 
283  using sumstat_type = typename std::result_of_t<StatisticsType(simulated_data_type&&)>;
284 
285  ReferenceTable<PriorPredictiveDistributionSample, param_type, sumstat_type> table;
286  unsigned int i = 0;
287  while(i < N)
288  {
289  auto params = param_type(priors(g));
290  try{
291  table.emplace_back(params, eta(generative_model(g, params)));
292  ++i;
293  }
294  catch (std::domain_error& e){
295  std::cerr << e.what() << std::endl;
296  }
297  }
298 
299  return table;
