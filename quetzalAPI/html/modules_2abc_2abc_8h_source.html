86  template<class ObsType, typename Generator>
87  std::vector<param_type> rubin_rejection_sampler(unsigned int N, ObsType const& observed_data, Generator& g) const {
88  std::vector<param_type> sample;

145  using simulated_data_type = typename std::result_of<ModelType(Generator&, const param_type&)>::type;
