132  template<
133  typename Merger, // = quetzal::coalescence::SimultaneousMultipleMerger<quetzal::coalescence::occupancy_spectrum::on_the_fly>,
134  typename Generator,
135  typename F,
136  typename Tree,
137  typename U
138  >
139  auto coalesce_along_spatial_history(forest_type<Tree> forest, F binary_op, Generator& gen, U make_tree) const
140  {
141 
142  this->check_consistency(m_history, forest);
143 
144  auto t = m_history.last_time();
145 
146  while( (forest.nb_trees() > 1) && (t > m_history.first_time()) )
147  {
148  may_coalesce_colocated<Merger>(forest, t, gen, binary_op, make_tree);
149  migrate_backward(forest, t, gen);
150  --t;
151  }
152  may_coalesce_colocated<Merger>(forest, t, gen, binary_op, make_tree);
153  return forest;
154  }

293  template<typename Merger, typename Generator, typename Tree, typename F, typename U>
294  void may_coalesce_colocated(forest_type<Tree>& forest, time_type const& t, Generator& gen, F binop, U make_tree) const noexcept
295  {
296  auto N = pop_size_history();
297 
298  for(auto const & x : forest.positions())
299  {
300  auto range = forest.trees_at_same_position(x);
301  std::vector<Tree> v;
302 
303  for(auto it = range.first; it != range.second; ++it)
304  {
305  v.push_back(it->second);
306  }
307 
308  if(v.size() >= 2){
309  auto last = Merger::merge(v.begin(), v.end(), N(x, t), make_tree(x,t), binop, gen );
310  forest.erase(x);
311  for(auto it = v.begin(); it != last; ++it){
312  forest.insert(x, *it);
313  }
314  }
315  }
316  }

335  /* @brief Coalesce a spatial forest of trees in a Wrigh-Fisher population until MRCA has been found.
336  *
337  * @tparam Space the coordinate type
338  * @tparam Tree the tree type
339  * @tparam Generator a random number Generator
340  * @tparam Binop a BranchingOperation functor for coalescence
341  * @tparam TimeFun an intialization functor for coalescence
342  *
343  * @param forest the spatial distribution of lineages to coalesce
344  * @param N number of gene copies in the Wright-Fisher population
345  * @param gen a random number Generator
346  * @param branch a branching binary operation (functor usually given by the selected coalescence policy).
347  * @param make_tree a initialization functor (functor usually given by the selected coalescence policy).
348  *
349  * @detail Coalesce lineages by sampling the waiting times directly in their distribution.
350  *
351  * @return One single tree.
352  */
353  template<typename Space, typename Tree, typename Generator, typename Binop, typename TimeFun>
354  static Tree coalesce(forest_type<Space, Tree> const& forest, unsigned int N, Generator& gen, Binop branch, TimeFun make_tree)
355  {
356 
357  if(forest.nb_trees() == 0){
358 
359  throw std::logic_error("The forest to coalesce is empty.");
360 
361  }else if(forest.nb_trees()==1){
362 
363  auto range = forest.trees_at_same_position(*(forest.positions().begin()));
364  assert(std::distance(range.first, range.second) == 1);
365  return range.first->second;
366 
367  }
368  assert(forest.nb_trees() >= 2 && "Trying to coalesce less than 2 nodes.");
369  auto trees = forest.get_all_trees();
370  return coalesce(trees, N, gen, branch, make_tree);
371  }
372 
373  /* @brief Simulates the number of generations to wait before the next coalescence event
374  *
375  * @tparam Generator a random number Generator
376  *
377  * @param k number of lineages
378  * @param N number of gene copies in the Wright-Fisher population
379  * @param gen a random number Generator
380  *
381  * @detail Samples the number of generations to wait before the next coalescence
382  * event in a geometric distribution of parameter {k \choose 2} / N
383  * (see Molecular evolution, a statistical approach, Z. Yang, 9.2 p 313).
384  *
385  * @return the number of generations to wait before the next coalescence event.
386  */
387  template<typename Generator>
388  static unsigned int sample_waiting_time(unsigned int k, unsigned int N, Generator& gen)
389  {
390  assert(N > k);
391  double mean = static_cast<double>(N) / boost::math::binomial_coefficient<double>(k, 2);
392  double p = 1 / (1 + mean);
393  assert( 0 < p && p <= 1);
394  std::geometric_distribution dist(p);
395  return dist(gen);
396  }

398  /*
399  * @brief Coalesce a non-spatial forest of trees in a Wrigh-Fisher population until MRCA has been found.
400  *
401  * @tparam Tree the tree type
402  * @tparam Generator a random number Generator
403  * @tparam Binop a BranchingOperation functor for coalescence
404  * @tparam TimeFun an intialization functor for coalescence
405  *
406  * @param trees the set of lineages to coalesce
407  * @param N number of gene copies in the Wright-Fisher population
408  * @param gen a random number Generator
409  * @param branch a branching binary operation (functor usually given by the selected coalescence policy).
410  * @param make_tree a initialization functor (functor usually given by the selected coalescence policy).
411  *
412  * @detail Coalesce lineages by sampling the waiting times directly in their
413  * distribution. The algorithm will perform one generation of simultaneous
414  * multiple merges if the number of lineages to coalesce k is superior to the
415  * population size N.
416  *
417  * @return One single tree.
418  */
419  template<typename Tree, typename Generator, typename Binop, typename TimeFun>
420  static Tree coalesce(std::vector<Tree> & trees, unsigned int N, Generator& gen, Binop branch, TimeFun make_tree)
421  {
422 
423  assert(trees.size() >= 2 && "Trying to coalesce less than 2 nodes.");
424 
425  if (trees.size() > N) // Wright Fisher assumptions are clearly violated
426  {
427  // Let's do one generation of simultaneous multiple merges to fix it.
428  unsigned int g = 1;
429  using smm_type = coalescence::SimultaneousMultipleMerger<coalescence::occupancy_spectrum::on_the_fly>;
430  auto last = smm_type::merge(trees.begin(), trees.end(), N, make_tree(g), branch, gen);
431  trees.erase(last, trees.end());
432  }
433 
434  unsigned int k = trees.size();
435  assert( k <= N && "number of lineages greater than ancestral Wright-Fisher population size.");
436  auto last = trees.end();
437 
438  while( k > 1 ){
439  unsigned int g = sample_waiting_time(k, N, gen);
440  last = quetzal::coalescence::binary_merge(trees.begin(), last, make_tree(g), branch, gen);
441  --k;
442  }
443 
444  return *(trees.begin());
445  }

448  /*
449  * @brief Coalesce a non-spatial forest of trees in a Wrigh-Fisher population
450  * during a finite number of generations.
451  *
452  * @tparam Tree the tree type
453  * @tparam Generator a random number Generator
454  * @tparam Binop a BranchingOperation functor for coalescence
455  * @tparam TimeFun an intialization functor for coalescence
456  *
457  * @param trees the set of lineages to coalesce
458  * @param N number of gene copies in the Wright-Fisher population
459  * @param gen a random number Generator
460  * g the number of generations that lasts the coalescence process
461  * @param branch a branching binary operation (functor usually given by the selected coalescence policy).
462  * @param make_tree a initialization functor (functor usually given by the selected coalescence policy).
463  *
464  * @detail Coalesce lineages by sampling the waiting times directly in their
465  * distribution. The algorithm will perform one generation of simultaneous
466  * multiple merges if the number of lineages to coalesce k is superior to the
467  * population size N.
468  *
469  * @return A set of trees, possibly of length 1 if MRCA was found during the g generations.
470  */
471  template<
472  typename MergerType = coalescence::SimultaneousMultipleMerger<coalescence::occupancy_spectrum::on_the_fly>,
473  typename Tree, typename Generator, typename Binop, typename TimeFun>
474  static std::vector<Tree> coalesce(std::vector<Tree> & trees, unsigned int N, unsigned int g, Generator& gen, Binop branch, TimeFun make_tree)
475  {
476  assert(trees.size() >= 2 && "Trying to coalesce less than 2 nodes.");
477  assert(g > 0 && "Number of generations for the coalescence process should be at least 1");
478  using merger_type = MergerType;
479  unsigned int t = 0;
480  auto last = trees.end();
481  while( t < g && std::distance(trees.begin(), last) > 1)
482  {
483  last = merger_type::merge(trees.begin(), last, N, make_tree(t), branch, gen);
484  ++t;
485  }
486  trees.erase(last, trees.end());
487  return trees;
488  }
