20  template<typename T>
21  auto make_neighboring_cells_functor(T const& landscape){
22 
23  using coord_type = quetzal::geography::GeographicCoordinates;
24  return [landscape](coord_type const& x0){
25  auto res = landscape.get().resolution();
26 
27  std::vector<coord_type> v;
28 
29  coord_type x1(x0);
30  x1.lon() += res.lon();
31  if(landscape.get().is_in_spatial_extent(x1)){
32  v.push_back(landscape.get().reproject_to_centroid(x1));
33  }
34 
35  coord_type x2(x0);
36  x2.lat() += res.lat();
37  if(landscape.get().is_in_spatial_extent(x2)){
38  v.push_back(landscape.get().reproject_to_centroid(x2));
39  }
40 
41  coord_type x3(x0);
42  x3.lon() -= res.lon();
43  if(landscape.get().is_in_spatial_extent(x3)){
44  v.push_back(landscape.get().reproject_to_centroid(x3));
45  }
46 
47  coord_type x4(x0);
48  x4.lat() -= res.lat() ;
49  if(landscape.get().is_in_spatial_extent(x4)){
50  v.push_back(landscape.get().reproject_to_centroid(x4));
51  }
52  return v;
53  };
54  };
