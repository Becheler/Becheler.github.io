80  template<typename F>
81  matrix_type make_weightm_matrix(std::vector<coord_type> const& coords, F w){
82  matrix_type A (coords.size(), coords.size());
83  for (unsigned i = 0; i < A.size1 (); ++ i)
84  {
85  auto x = coords.at(i);
86  for (unsigned j = 0; j < A.size2(); ++ j)

103  for (unsigned i = 0; i < m_matrix.size1 (); ++ i)
104  {

106  weights.reserve(m_matrix.size2());
107  for (unsigned j = 0; j < m_matrix.size2(); ++ j)
108  {
109  weights.push_back(m_matrix(i, j));
110  }
111  dists.emplace_back(weights.cbegin(), weights.cend());

243  template<typename F>

246  matrix_type A (coords.size(), coords.size());

252  auto y = coords.at(j);

257  }

266  template<typename F>
267  Implementation(std::vector<point_ID_type> const& points, std::vector<coord_type> const& coords, F f) :

269  m_points(points),
270  m_coords(coords)
271  {}
279  auto arrival_space(coord_type const& x) const
280  {
281  auto it = m_arrival_space_cash.find(x);
282  if(it != m_arrival_space_cash.end())
283  {
284  return it->second;
285  }
286  else
287  {
288  auto p = m_arrival_space_cash.emplace(x, retrieve_non_zero_arrival_space(x));
289  return p.first->second;
290  }
291  }
295  auto retrieve_non_zero_arrival_space(coord_type const& x) const
296  {
297  using it1_t = typename matrix_type::const_iterator1;
298  using it2_t = typename matrix_type::const_iterator2;
299  auto i = getIndexOfPointInVector(x, m_coords);
300  it1_t it1 = m_matrix.begin1();
301  std::advance(it1, i);
302  std::vector<coord_type> v;
303  v.reserve(m_matrix.size2()); // worst case
304  for (it2_t it2 = it1.begin(); it2 != it1.end(); it2++)
305  {
306  if(*it2 > 0.0) v.push_back(m_coords.at(it2.index2()));
307  }
308  return v;
309  }
313  double migration_rate(coord_type const& x, coord_type const& y)
314  {
315  return m_matrix(getIndexOfPointInVector(x, m_coords), getIndexOfPointInVector(y, m_coords) );
316  }

437  std::vector<coord_type> arrival_space(coord_type const& x) const
438  {
439  auto v = m_get_neighbors(x);
440  // let's add focal deme to have all possible arrival demes
441  v.push_back(x);
442  return v;
443  }
444 
448  auto operator()(coord_type const& from, coord_type const& to)
449  {
450  if(from == to)
451  {
452  return 1.0 - m_emigrant_rate;
453  }
454  else
455  {
456  return m_emigrant_rate * friction_weights(from, to);
457  }
458  }
459  private:
460  double friction_weights(coord_type const& x, coord_type const& y) const
461  {
462  // TODO does it copy the whole vector?
463  auto neighbors = m_get_neighbors(x);
464  std::vector<double> v(neighbors.size(), 0.0);
465  std::transform(neighbors.begin(), neighbors.end(), v.begin(), [this](coord_type const& z) { return 1.0/(this->m_friction(z));} );
466  double sum = std::accumulate(v.begin(), v.end(), 0.0);
467  return 1.0/(m_friction(y)*sum);
468  }
469  }; // end neighboring_migration
