87  constrained_sampling(std::vector<coord_type> const& space, F1 f, N pop_size, unsigned int n) :
88  _weights(compute_weights(space, f)),

98  void param(unsigned int n){ _param = param_type(n); }
99 
100  /* @brief Sample points in a discrete space, uniformely at random
101  * @remark a same coordinates can be sampled several times
102  *
103  * @tparam Cont a container of coordinates
104  * @tparam Generator a RandomNumberGenerator
105  *
106  * @param space coordinates to be sampled
107  * @param n the number of points to sample
108  * @param gen a random number generator
109  *
110  * @return a 'std::map<Cont::value_type, unsigned int>' giving the sampling intensities
111  * at the sampled coordinates.
112  */
113  template<typename Generator>
114  auto operator()(Generator& gen) const
115  {
116  std::discrete_distribution<size_t> dist(_weights.begin(), _weights.end());
117  std::map<coord_type, unsigned int> sample;
118  unsigned int i = 0;
119  unsigned int nb_try = 0;
120  while(i < _param.sampling_size() && nb_try < 10 * (_param.sampling_size()) ){
121  auto id = dist(gen);
122  if(sample[_space[id]] < _population_size(_space[id]) ){
123  sample[_space[id]] += 1;
124  ++i;
125  }
126  ++nb_try;
127  }
128  if(sample.empty()){
129  throw std::logic_error("Sampling scheme unable to populate a sample.");
130  }
131  return sample;
132  }
133 
134 }; // constrained_sampling
