138  template<typename Generator, typename F, typename Tree>
139  forest_type<Tree> coalesce_along_history(forest_type<Tree> forest, F binary_op, Generator& gen) const
140  {
141 
142  this->check_consistency(m_history, forest);
143 
144  auto t = m_history.last_time();
145 
146  while( (forest.nb_trees() > 1) && (t > m_history.first_time()) )
147  {
148  may_coalesce_colocated(forest, t, gen, binary_op);
149  migrate_backward(forest, t, gen);
150  --t;
151  }
152  may_coalesce_colocated(forest, t, gen, binary_op);

154  }

158  template<typename Generator, typename Forest>
159  void migrate_backward(Forest & forest, time_type t, Generator& gen) const noexcept
160  {
161  Forest new_forest;
162  for(auto const it : forest)
163  {
164  coord_type x = it.first;
165  new_forest.insert(m_history.backward_kernel(x, t, gen), it.second);
166  }
167  assert(forest.nb_trees() == new_forest.nb_trees());
168  forest = new_forest;
169  }

172  template<typename Generator, typename Tree, typename F>
173  void may_coalesce_colocated(forest_type<Tree>& forest, time_type const& t, Generator& gen, F binop) const noexcept
174  {
175  auto N = pop_size_history();
176 
177  for(auto const & x : forest.positions())
178  {
179  auto range = forest.trees_at_same_position(x);
180  std::vector<Tree> v;
181 
182  for(auto it = range.first; it != range.second; ++it)
183  {
184  v.push_back(it->second);
185  }
186 
187  if(v.size() >= 2){
188  auto last = Policy<strategy_type>::merger_type::merge(v.begin(), v.end(), N(x, t), Tree(), binop, gen );
189  forest.erase(x);
190  for(auto it = v.begin(); it != last; ++it){
191  forest.insert(x, *it);
192  }
193  }
194  }
195  }
