44  template<typename Generator, typename Forest>
45  void simulate_backward_migration(Forest & forest, time_type const& t, Generator& gen) const
46  {
47  Forest new_forest;
48  for(auto const it : forest)

50  coord_type x = it.first;

57  template<typename Generator, typename Tree>
58  void coalesce(forest_type<Tree>& forest, time_type const& t, Generator& gen) const
59  {
60  auto const& N = m_history.N();
61 
62  for(auto const & x : forest.positions())
63  {
64  auto range = forest.trees_at_same_position(x);
65  std::vector<Tree> v;
66 
67  for(auto it = range.first; it != range.second; ++it)
68  {
69  v.push_back(it->second);
70  }
71 
72  if(v.size() >= 2){
73  auto last = merger_type::merge(v.begin(), v.end(), N(x, t), gen );
74  forest.erase(x);
75  for(auto it = v.begin(); it != last; ++it){
76  forest.insert(x, *it);
77  }
78  }
79 
80  }
81  }

116  template<typename Tree>
117  bool is_history_consistent_with_sampling(history_type const& history, forest_type<Tree> const& forest) const
118  {
119  time_type t = history.last_time();
120  for(auto const& it : forest.positions())
121  {
122  if( !history.N().is_defined(it, t) || history.N()(it, t) < forest.nb_trees(it))
123  {
124  return false;
125  }
126  }
127  return true;

146  template<typename Generator, typename Tree>
147  forest_type<Tree> coalescence_process(forest_type<Tree> forest, history_type const& history, Generator& gen)
148  {
149  auto t = history.last_time();
150  while( (forest.nb_trees() > 1) && (t > history.first_time()) )
151  {
152  coalesce(forest, t, gen);
153  simulate_backward_migration(forest, t, gen);
154  --t;
155  }
156  coalesce(forest, t, gen);
157  return forest;
158  }
