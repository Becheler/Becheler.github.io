142  forest_type<Tree> coalesce_along_spatial_history(forest_type<Tree> forest, F binary_op, Generator& gen, U make_tree) const
143  {

184  template<typename Generator, typename Tree, typename F, typename U>
185  void may_coalesce_colocated(forest_type<Tree>& forest, time_type const& t, Generator& gen, F binop, U make_tree) const noexcept
186  {
187  auto N = pop_size_history();
188 
189  for(auto const & x : forest.positions())
190  {
191  auto range = forest.trees_at_same_position(x);
192  std::vector<Tree> v;
193 
194  for(auto it = range.first; it != range.second; ++it)
195  {
196  v.push_back(it->second);
197  }
198 
199  if(v.size() >= 2){
200  auto last = DemographicAssumptionPolicy<strategy_type>::merger_type::merge(v.begin(), v.end(), N(x, t), make_tree(x,t), binop, gen );
201  forest.erase(x);
202  for(auto it = v.begin(); it != last; ++it){
203  forest.insert(x, *it);
204  }
205  }
206  }

211 class DiscreteTimeWrightFisher
212 {
213 
214 public:

216  template<typename Space, typename Tree>
217  using forest_type = quetzal::coalescence::Forest<Space, Tree>;
218 
219  // @brief Simulates the number of generations to wait before the next coalescence event
220  //
221  // @tparam Generator a random number Generator
222  // @param k number of lineages
223  // @param N number of gene copies in the Wright-Fisher population
224  // @gen a random number Generator
225  //
226  // Samples the number of generations to wait before the next coalescence event in
227  // a geometric distribution of parameter {k \choose 2} / N
228  //
229  //
230  // @return the number of generations to wait before the next coalescence event.
231  //
232  template<typename Generator>
233  static unsigned int sample_waiting_time(unsigned int k, unsigned int N, Generator& gen)
234  {
235  assert(N > k);
236  // Molecular evolution, a statistical approach, Z. Yang, 9.2 p 313
237  double mean = static_cast<double>(N) / boost::math::binomial_coefficient<double>(k, 2);
238  double p = 1 / (1 + mean);
239  assert( 0 < p && p <= 1);
240  std::geometric_distribution dist(p);
241  return dist(gen);
242  }
243

245  template<typename Tree, typename Generator, typename Binop, typename TimeFun>
246  static Tree coalesce(std::vector<Tree> & trees, unsigned int N, Generator& gen, Binop branch, TimeFun make_tree)
247  {
248  auto last = trees.end();
249  unsigned int k = std::distance(trees.begin(), last);
250  assert(k > 1);
251  while( k > 1 ){
252  unsigned int g = sample_waiting_time(k, N, gen);
253  last = quetzal::coalescence::binary_merge(trees.begin(), last, make_tree(g), branch, gen);
254  --k;
255  }
256 
257  return *(trees.begin());
258  }

261  template<typename Space, typename Tree, typename Generator, typename Binop, typename TimeFun>
262  static Tree coalesce(forest_type<Space, Tree> const& forest, unsigned int N, Generator& gen, Binop branch, TimeFun make_tree)
263  {
264 
265  if(forest.nb_trees()==1){
266  auto range = forest.trees_at_same_position(*(forest.positions().begin()));
267  assert(std::distance(range.first, range.second) == 1);
268  return range.first->second;
269  }
270 
271  std::vector<Tree> trees;
272  trees.reserve(forest.nb_trees());
273  for(auto & it : forest){
274  trees.push_back(it.second);
275  }
276 
277  return coalesce(trees, N, gen, branch, make_tree);
