45 struct on_the_fly {
46  template<typename Generator>
47  static auto sample(unsigned int k, unsigned int N, Generator& g){
48 
49  std::vector<unsigned int> M_j(k+1);
50  std::vector<unsigned int> parents(N);
51 
52  std::uniform_int_distribution<unsigned int> dist(0,N-1);
53  for(unsigned int i = 1; i <=k; ++i){
54  parents[dist(g)] +=1;
55  }
56 
57  for(unsigned int it : parents){
58  M_j[it] += 1;
59  }
60 
61  assert(M_j.size() == k+1);
62  assert(test_number_of_balls_conservation(k, M_j));
63  assert(test_number_of_urns_conservation(N, M_j));
64 
65  return M_j;
66  }
