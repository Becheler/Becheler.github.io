Detailed Description

template<class SpectrumCreationPolicy>
struct quetzal::coalescence::SimultaneousMultipleMerger< SpectrumCreationPolicy >

Discrete generation simultaneous multiple merger.

Defines a policy class to coalesce multiple nodes selected uniformely at random in a range, according to an OccupancySpectrum. Its interface makes it exchangeable with other merger policies.

Template Parameters
SpectrumCreationPolicyThe policy used to generate an OccupancySpectrum.

Example

#include <vector>
#include <list>
#include <string>
#include <algorithm>
#include <iostream>
#include <iterator>
#include <random>
#include <cassert>
int main () {
using node_type = std::string;
using quetzal::coalescence::occupancy_spectrum::on_the_fly;
using SMM = quetzal::coalescence::SimultaneousMultipleMerger<on_the_fly> ;
std::vector<node_type> nodes = {"a", "b", "c", "d"};
unsigned int N = 3;
std::mt19937 gen;
std::cout << "\nNodes at sampling time :\n";
std::copy(nodes.begin(), nodes.end(), std::ostream_iterator<node_type>(std::cout, "\n"));
std::cout << "\n";
auto last = SMM::merge(nodes.begin(), nodes.end(), N, gen);
std::cout << "\nAfter one simultaneous multiple merge generation:\n";
for(auto it = nodes.begin(); it != last; ++it){
std::cout << *it << std::endl;
}
return 0;
}
